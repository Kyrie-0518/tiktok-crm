# ── 虾掌柜ERP 宿主 Nginx 配置 ──
# 部署路径：/etc/nginx/sites-available/bozone  →  /etc/nginx/sites-enabled/bozone

# HTTP → HTTPS 重定向
server {
    listen 80;
    listen [::]:80;
    server_name bvefdvp.cn www.bvefdvp.cn;

    # Let's Encrypt 验证路径
    location /.well-known/acme-challenge/ {
        root /var/www/html;
    }

    # 其他请求强制跳转 HTTPS
    location / {
        return 301 https://$host$request_uri;
    }
}

# HTTPS 服务
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name bvefdvp.cn;

    # SSL 证书（certbot 自动生成）
    ssl_certificate     /etc/letsencrypt/live/bvefdvp.cn/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/bvefdvp.cn/privkey.pem;

    # SSL 优化
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;

    # 日志
    access_log /var/log/nginx/bozone.access.log;
    error_log  /var/log/nginx/bozone.error.log;

    # 上传限制
    client_max_body_size 100m;

    # ── API 反向代理到后端容器 ──
    location /api/ {
        proxy_pass http://127.0.0.1:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 300s;
        proxy_send_timeout 300s;
    }

    # ── 上传文件 ──
    location /uploads/ {
        proxy_pass http://127.0.0.1:3000;
        proxy_set_header Host $host;
        proxy_buffering off;
        client_max_body_size 100m;
    }

    # ── 前端静态资源（容器内 Nginx） ──
    location / {
        proxy_pass http://127.0.0.1:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

/**
 * Pipeline Orchestrator — 串联全部 8 个阶段
 * Data Collector → Mapper → Rule Engine → Evidence Builder → Diagnosis → Suggestions → Report → Export
 */
import getDb from '../../db';
import { collectAllData } from './collector';
import { runRules } from './rules';
import { buildEvidence } from './evidence';
import { runDiagnosis } from './diagnosis';
import { DiagnosisResult } from './types';

export async function runPipeline(shopCipher: string, shopName: string, days = 30): Promise<DiagnosisResult> {
  const taskId = `dx-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`;
  const db = getDb();

  // 更新状态：running
  db.prepare(`INSERT OR REPLACE INTO growth_diagnoses (task_id, shop_id, shop_name, status, created_at, updated_at)
    VALUES (?, ?, ?, 'running', datetime('now','localtime'), datetime('now','localtime'))`)
    .run(taskId, shopCipher, shopName);

  try {
    // Stage 1: Data Collector
    const data = await collectAllData(shopCipher, days);

    // Stage 2: Data Mapper (内嵌在 collector 中)

    // Stage 3: Rule Engine
    const rules = runRules(data);

    // Stage 4: Evidence Builder
    const { evidence, risks } = buildEvidence(data, rules);

    // Stage 5-6: AI Diagnosis + Suggestions
    const diag = await runDiagnosis(shopName, data, rules, evidence);

    // Stage 7: Build final report
    const result: DiagnosisResult = {
      task_id: taskId,
      shop_id: shopCipher,
      shop_name: shopName,
      created_at: new Date().toISOString(),
      health_score: diag.health_score,
      health_grade: diag.health_grade,
      executive_summary: diag.executive_summary,
      problems: diag.problems,
      risk_ranking: risks,
      evidence_chains: evidence,
      suggestions: diag.suggestions,
      action_plan: diag.action_plan,
      data_snapshot: data,
    };

    // Save to DB
    db.prepare(`UPDATE growth_diagnoses SET
      status = 'completed', health_score = ?, health_grade = ?,
      executive_summary = ?, result_json = ?, updated_at = datetime('now','localtime')
      WHERE task_id = ?`)
      .run(diag.health_score, diag.health_grade, diag.executive_summary,
           JSON.stringify(result), taskId);

    return result;
  } catch (e: any) {
    db.prepare(`UPDATE growth_diagnoses SET status = 'failed', error_message = ?, updated_at = datetime('now','localtime') WHERE task_id = ?`)
      .run(e.message, taskId);
    throw e;
  }
}

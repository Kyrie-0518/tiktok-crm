import React, { useState, useEffect } from 'react';
import { Card, Button, Typography, Select, message, Statistic, Row, Col, Spin, Empty, Space, Progress, Tag } from 'antd';
import { ArrowUpOutlined, ArrowDownOutlined, ThunderboltOutlined, BarChartOutlined } from '@ant-design/icons';
import api from '../../api';
const { Text, Title, Paragraph } = Typography;

const DS = { bg: '#f5f3f0', cardBg: '#FFFFFF', cardBorder: '#e8e5e0', primary: '#2563eb', text: '#1A1A2E', textSecondary: '#6B7280', radius: 12 };

export default function AIReview() {
  const [diagnoses, setDiagnoses] = useState<any[]>([]);
  const [prevId, setPrevId] = useState<string>('');
  const [currId, setCurrId] = useState<string>('');
  const [review, setReview] = useState<any>(null);
  const [reviewing, setReviewing] = useState(false);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    api.get('/growth-center/history', { params: { limit: 20 } }).then(r => {
      const completed = (r.data?.rows || []).filter((d: any) => d.status === 'completed');
      setDiagnoses(completed);
      if (completed.length >= 2) { setPrevId(completed[1].task_id); setCurrId(completed[0].task_id); }
    }).catch(() => {}).finally(() => setLoading(false));
  }, []);

  const runReview = async () => {
    if (!prevId || !currId) return message.warning('请选择两次诊断');
    setReviewing(true);
    try {
      const res = await api.post('/growth-center/review', { previous_task_id: prevId, current_task_id: currId });
      setReview(res.data);
    } catch (e: any) {
      message.error('复盘失败');
    } finally { setReviewing(false); }
  };

  if (loading) return <div style={{ textAlign: 'center', padding: 80 }}><Spin /></div>;
  if (diagnoses.length < 2) return (
    <div style={{ padding: 24, background: DS.bg, minHeight: '100%' }}>
      <Card style={{ borderRadius: DS.radius, textAlign: 'center', padding: 60 }}>
        <Empty description="需要至少 2 次诊断记录才能进行 AI 复盘" image={Empty.PRESENTED_IMAGE_SIMPLE}>
          <Text type="secondary">请先完成两次店铺诊断</Text>
        </Empty>
      </Card>
    </div>
  );

  return (
    <div style={{ padding: 24, background: DS.bg, minHeight: '100%' }}>
      <div style={{ display: 'flex', alignItems: 'center', gap: 12, marginBottom: 24 }}>
        <div style={{ width: 42, height: 42, borderRadius: 12, background: `linear-gradient(135deg, #8B5CF6, #A78BFA)`, display: 'flex', alignItems: 'center', justifyContent: 'center', color: '#fff', fontSize: 20 }}><BarChartOutlined /></div>
        <div style={{ flex: 1 }}>
          <Title level={4} style={{ margin: 0, fontSize: 20, fontWeight: 700, color: DS.text }}>AI 复盘</Title>
          <Text type="secondary" style={{ fontSize: 13 }}>对比两次诊断结果，分析优化效果</Text>
        </div>
      </div>

      <Card style={{ borderRadius: DS.radius, border: `1px solid ${DS.cardBorder}`, marginBottom: 16 }}>
        <Space size={16}>
          <Select placeholder="上次诊断" value={prevId} onChange={setPrevId} style={{ width: 300 }}
            options={diagnoses.map(d => ({ value: d.task_id, label: `${d.shop_name || ''} - ${d.created_at ? new Date(d.created_at).toLocaleDateString('zh-CN') : ''}` }))} />
          <Select placeholder="本次诊断" value={currId} onChange={setCurrId} style={{ width: 300 }}
            options={diagnoses.map(d => ({ value: d.task_id, label: `${d.shop_name || ''} - ${d.created_at ? new Date(d.created_at).toLocaleDateString('zh-CN') : ''}` }))} />
          <Button type="primary" icon={<ThunderboltOutlined />} onClick={runReview} loading={reviewing} style={{ borderRadius: 8, background: '#8B5CF6' }}>开始复盘</Button>
        </Space>
      </Card>

      {review && (
        <Row gutter={16}>
          <Col span={8}>
            <Card style={{ borderRadius: DS.radius, border: `1px solid ${DS.cardBorder}`, textAlign: 'center', padding: 24 }}>
              <Text type="secondary">距离上次诊断</Text>
              <div style={{ fontSize: 36, fontWeight: 700, color: DS.text, margin: '8px 0' }}>{review.days_between}<Text style={{ fontSize: 16, fontWeight: 400 }}> 天</Text></div>
            </Card>
          </Col>
          <Col span={8}>
            <Card style={{ borderRadius: DS.radius, border: `1px solid ${DS.cardBorder}`, textAlign: 'center', padding: 24 }}>
              <Text type="secondary">健康评分变化</Text>
              <div style={{ fontSize: 36, fontWeight: 700, color: review.health_change > 0 ? '#10B981' : review.health_change < 0 ? '#EF4444' : DS.text, margin: '8px 0' }}>
                {review.health_change > 0 ? '+' : ''}{review.health_change}
              </div>
            </Card>
          </Col>
          <Col span={8}>
            <Card style={{ borderRadius: DS.radius, border: `1px solid ${DS.cardBorder}`, textAlign: 'center', padding: 24 }}>
              <Text type="secondary">GMV 变化</Text>
              <div style={{ fontSize: 36, fontWeight: 700, color: DS.text, margin: '8px 0' }}>
                {review.gmv_change_pct > 0 ? '+' : ''}{review.gmv_change_pct}%
              </div>
            </Card>
          </Col>
          <Col span={24} style={{ marginTop: 16 }}>
            <Card style={{ borderRadius: DS.radius, border: `1px solid ${DS.cardBorder}` }}>
              <Text strong style={{ display: 'block', marginBottom: 8 }}>AI 复盘总结</Text>
              <Paragraph type="secondary">{review.ai_summary}</Paragraph>
            </Card>
          </Col>
        </Row>
      )}
    </div>
  );
}

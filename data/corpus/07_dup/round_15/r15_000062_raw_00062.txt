// Video Adapter — 统一模型调用参数
import { AdapterInput, AdapterOutput } from '../types';

const MODEL_CONFIGS: Record<string, { baseParams: Record<string, any>; tokenEstimate: (duration: number, count: number) => number }> = {
  seedance: { baseParams: { model: 'doubao-seedance-2-0-260128', version: 'v2' }, tokenEstimate: (d, c) => d * c * 12 },
  kling: { baseParams: { model: 'kling-2.1', mode: 'pro' }, tokenEstimate: (d, c) => d * c * 15 },
  minimax: { baseParams: { model: 'minimax-video-1', quality: 'high' }, tokenEstimate: (d, c) => d * c * 10 },
};

export function adapterAgent(input: AdapterInput): AdapterOutput {
  const modelLower = input.model.toLowerCase();
  let adapterKey = 'seedance';
  if (modelLower.includes('kling')) adapterKey = 'kling';
  else if (modelLower.includes('minimax')) adapterKey = 'minimax';

  const cfg = MODEL_CONFIGS[adapterKey] || MODEL_CONFIGS.seedance;

  return {
    generateParams: {
      ...cfg.baseParams,
      prompt: input.optimizer.optimizedPrompts[0]?.prompt,
      negative_prompt: input.optimizer.optimizedPrompts[0]?.negativePrompt,
      resolution: input.resolution,
      aspect_ratio: input.aspectRatio,
      duration: input.duration,
      count: input.count,
    },
    modelName: cfg.baseParams.model,
    estimatedTokens: cfg.tokenEstimate(input.duration, input.count),
  };
}

import { useEffect, useState } from 'react';
import {
  Card, Button, Space, Tag, Modal, message, Input, Select,
  Typography, Row, Col, Popconfirm, Alert
} from 'antd';
import {
  VideoCameraOutlined, DownloadOutlined, ShareAltOutlined,
  DeleteOutlined, SearchOutlined, ReloadOutlined,
  PlayCircleOutlined
} from '@ant-design/icons';
import api from '../api';
import { formatDateTimeSec } from '../utils/time';
import DataTable from '../components/DataTable';
import { useIsDeveloper } from '../stores/authStore';

const { Text } = Typography;

interface Video {
  id: number;
  title: string;
  video_url: string;
  thumbnail_url: string;
  prompt: string;
  model: string;
  model_type: string;
  resolution: string;
  duration: number;
  aspect_ratio: string;
  status: string;
  product_name?: string;
  product_image?: string;
  created_at: string;
  user_id?: number;
  username?: string;  // developer 模式下返回
}

export default function MaterialLibrary() {
  const isDeveloper = useIsDeveloper();
  const [videos, setVideos] = useState<Video[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [previewVideo, setPreviewVideo] = useState<Video | null>(null);
  const [filterStatus, setFilterStatus] = useState<string>('all');
  const [searchKeyword, setSearchKeyword] = useState('');

  const loadVideos = async () => {
    setLoading(true);
    setError(null);
    try {
      const res = await api.get('/seedance/videos');
      setVideos(res.data);
    } catch (e: any) {
      const msg = e.response?.data?.error || e.message || '未知错误';
      message.error('加载素材库失败: ' + msg);
      setError(msg);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    loadVideos();
  }, []);

  const handleDelete = async (id: number) => {
    try {
      await api.delete(`/seedance/videos/${id}`);
      message.success('删除成功');
      loadVideos();
    } catch (e: any) {
      message.error('删除失败: ' + (e.response?.data?.error || e.message));
    }
  };

  const filteredVideos = videos.filter(v => {
    if (filterStatus !== 'all' && v.status !== filterStatus) return false;
    if (searchKeyword && !v.title?.includes(searchKeyword) && !v.prompt?.includes(searchKeyword)
      && !(isDeveloper && v.username?.toLowerCase().includes(searchKeyword.toLowerCase()))) return false;
    return true;
  });

  const columns = [
    {
      title: '预览',
      key: 'preview',
      width: 100,
      render: (_: any, r: Video) => {
        const hasValidThumb = r.thumbnail_url && !r.thumbnail_url.includes('product') && !r.thumbnail_url.includes('image');
        return (
          <div
            style={{
              width: 84, height: 56,
              background: '#f5f5f5',
              borderRadius: 8, overflow: 'hidden',
              cursor: 'pointer',
              display: 'flex', alignItems: 'center', justifyContent: 'center',
              position: 'relative',
              border: '1px solid #f0f0f0',
            }}
            onClick={() => setPreviewVideo(r)}
          >
            {hasValidThumb ? (
              <img src={r.thumbnail_url} style={{ width: '100%', height: '100%', objectFit: 'cover' }} alt="" />
            ) : r.video_url ? (
              <video src={r.video_url} muted preload="metadata"
                style={{ width: '100%', height: '100%', objectFit: 'cover' }}
                onError={(e) => { (e.target as HTMLVideoElement).style.display = 'none'; }}
              />
            ) : (
              <VideoCameraOutlined style={{ fontSize: 20, color: '#ccc' }} />
            )}
            {r.status === 'completed' && (
              <div style={{
                position: 'absolute', inset: 0,
                display: 'flex', alignItems: 'center', justifyContent: 'center',
                background: 'rgba(15,23,42,0.25)', borderRadius: 7,
              }}>
                <PlayCircleOutlined style={{ fontSize: 24, color: 'rgba(255,255,255,0.95)' }} />
              </div>
            )}
          </div>
        );
      },
    },
    {
      title: '标题 / 提示词',
      key: 'info',
      render: (_: any, r: Video) => (
        <div style={{ minWidth: 0 }}>
          <Text strong style={{ fontSize: 13, color: '#333' }} ellipsis>
            {r.title || '无标题'}
          </Text>
          <div style={{ marginTop: 2 }}>
            <Text type="secondary" style={{ fontSize: 12, maxWidth: 240 }} ellipsis>
              {r.prompt?.substring(0, 60)}{r.prompt?.length > 60 ? '...' : ''}
            </Text>
          </div>
        </div>
      ),
    },
    {
      title: '模型',
      dataIndex: 'model',
      width: 200,
      ellipsis: true,
      render: (v: string) => {
        if (!v) return <span style={{ color: '#ccc' }}>-</span>;
        const name = v.replace(/^[\w.-]+:\s*/, '');
        return (
          <span style={{
            fontFamily: '"SF Mono", Consolas, Monaco, monospace',
            fontSize: 12, color: '#666', fontWeight: 500,
            letterSpacing: -0.2,
          }}>
            {name}
          </span>
        );
      },
    },
    {
      title: '规格',
      key: 'specs',
      width: 130,
      render: (_: any, r: Video) => (
        <Text type="secondary" style={{ fontSize: 12 }}>
          {r.resolution} · {r.duration}s
        </Text>
      ),
    },
    {
      title: '状态',
      dataIndex: 'status',
      width: 90,
      render: (v: string) => {
        const config: Record<string, { color: string; text: string }> = {
          completed: { color: 'success', text: '已完成' },
          generating: { color: 'processing', text: '生成中' },
          failed: { color: 'error', text: '失败' },
        };
        const c = config[v] || { color: 'default', text: v };
        return <Tag color={c.color} style={{ borderRadius: 6, fontWeight: 500, fontSize: 12 }}>{c.text}</Tag>;
      },
    },
    ...(isDeveloper ? [{
      title: '用户',
      dataIndex: 'username',
      width: 90,
      render: (v: string) => <span style={{ fontSize: 13, color: '#2563eb', fontWeight: 500 }}>{v || '-'}</span>,
    }] : []),
    {
      dataIndex: 'created_at',
      width: 160,
      render: (v: string) => {
        if (!v) return '-';
        return formatDateTimeSec(v);
      },
    },
    {
      title: '操作',
      key: 'actions',
      width: 160,
      render: (_: any, r: Video) => (
        <Space size={4}>
          {r.status === 'completed' && r.video_url && (
            <>
              <Button type="link" size="small" icon={<DownloadOutlined />} onClick={() => window.open(r.video_url, '_blank')} style={{ fontSize: 12 }}>
                下载
              </Button>
              <Button type="link" size="small" icon={<ShareAltOutlined />} onClick={() => {
                navigator.clipboard.writeText(r.video_url);
                message.success('链接已复制');
              }} style={{ fontSize: 12 }}>
                分享
              </Button>
            </>
          )}
          <Popconfirm title="确定删除？" onConfirm={() => handleDelete(r.id)}>
            <Button type="link" danger size="small" icon={<DeleteOutlined />} style={{ fontSize: 12 }} />
          </Popconfirm>
        </Space>
      ),
    },
  ];

  return (
    <div style={{ padding: '20px 24px', background: '#f5f3f0', minHeight: '100%' }}>
      {/* 页面标题栏 */}
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', marginBottom: 16 }}>
        <div style={{ display: 'flex', alignItems: 'center', gap: 10 }}>
          <div style={{
            width: 36, height: 36, borderRadius: 10,
            background: 'linear-gradient(135deg, #6366f1 0%, #8b5cf6 100%)',
            display: 'flex', alignItems: 'center', justifyContent: 'center',
            color: '#fff', fontSize: 18,
          }}>
            <VideoCameraOutlined />
          </div>
          <div>
            <h2 style={{ margin: 0, fontSize: 18, fontWeight: 700, color: '#1e293b' }}>素材库</h2>
            <span style={{ fontSize: 12, color: '#999' }}>AI 视频生成记录管理</span>
          </div>
        </div>

        <Space size={12}>
          <Input
            placeholder="搜索标题或提示词..."
            prefix={<SearchOutlined style={{ color: '#999' }} />}
            style={{ width: 220, borderRadius: 8 }}
            value={searchKeyword}
            onChange={e => setSearchKeyword(e.target.value)}
            allowClear
          />
          <Select
            value={filterStatus}
            onChange={setFilterStatus}
            style={{ width: 100, borderRadius: 8 }}
            options={[
              { label: '全部状态', value: 'all' },
              { label: '已完成', value: 'completed' },
              { label: '生成中', value: 'generating' },
              { label: '失败', value: 'failed' },
            ]}
          />
          <Button icon={<ReloadOutlined />} onClick={loadVideos} style={{ borderRadius: 8, fontWeight: 500 }}>
            刷新
          </Button>
        </Space>
      </div>

      {/* 表格卡片 */}
      {error && (
        <Alert type="error" message="加载失败" description={error} showIcon closable
          style={{ marginBottom: 12, borderRadius: 10 }} />
      )}
      <Card
        bordered={false}
        bodyStyle={{ padding: 0 }}
        style={{ borderRadius: 12, background: '#ffffff', border: '1px solid #e8e5e0' }}
      >
        <DataTable dataSource={filteredVideos} columns={columns} loading={loading} />
      </Card>

      {/* 视频预览弹窗 */}
      <Modal
        title={
          <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
            <PlayCircleOutlined style={{ color: '#6366f1', fontSize: 18 }} />
            <span>视频预览</span>
          </div>
        }
        open={!!previewVideo}
        onCancel={() => setPreviewVideo(null)}
        footer={null}
        width={720}
        styles={{
          body: { padding: '20px 0' },
          header: { borderBottom: '1px solid #f0f0f0', padding: '16px 24px' },
        }}
      >
        {previewVideo && (
          <div style={{ textAlign: 'center', padding: '0 24px' }}>
            <video src={previewVideo.video_url} controls autoPlay
              style={{ width: '100%', maxHeight: 420, borderRadius: 10, background: '#000' }}
            />
            <Row gutter={16} style={{ marginTop: 18 }}>
              <Col span={12}>
                <Text type="secondary" style={{ fontSize: 13 }}>模型：</Text>
                <span style={{ fontFamily: '"SF Mono", Consolas, monospace', fontSize: 13, color: '#666', fontWeight: 500 }}>
                  {(previewVideo.model || '').replace(/^[\w.-]+:\s*/, '')}
                </span>
              </Col>
              <Col span={12}>
                <Text type="secondary" style={{ fontSize: 13 }}>规格：</Text>
                <Text style={{ fontSize: 13, color: '#666' }}>{previewVideo.resolution} · {previewVideo.duration}s · {previewVideo.aspect_ratio}</Text>
              </Col>
            </Row>
            <div style={{ marginTop: 14, textAlign: 'left' }}>
              <Text type="secondary" style={{ fontSize: 13 }}>提示词：</Text>
              <div style={{
                marginTop: 6, padding: '10px 14px',
                background: '#f5f5f5', borderRadius: 8,
                border: '1px solid #f0f0f0', lineHeight: 1.7,
              }}>
                <Text style={{ fontSize: 13, color: '#666' }}>{previewVideo.prompt || '无'}</Text>
              </div>
            </div>
            <Space style={{ marginTop: 18 }} size={12}>
              {previewVideo.video_url && (
                <Button type="primary" icon={<DownloadOutlined />}
                  onClick={() => window.open(previewVideo.video_url, '_blank')}
                  style={{ borderRadius: 8, fontWeight: 500 }}>
                  下载视频
                </Button>
              )}
              <Button icon={<ShareAltOutlined />}
                onClick={() => { navigator.clipboard.writeText(previewVideo.video_url); message.success('链接已复制'); }}
                style={{ borderRadius: 8 }}>
                复制链接
              </Button>
            </Space>
          </div>
        )}
      </Modal>
    </div>
  );
}

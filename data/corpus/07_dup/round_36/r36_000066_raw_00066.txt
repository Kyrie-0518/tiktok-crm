// AI Quality Agent — 生成后质量评估
import { callLLM, parseJSON } from '../llm';
import { QualityInput, QualityOutput } from '../types';

export async function qualityAgent(input: QualityInput): Promise<QualityOutput> {
  // 如果没有视频URL，只评估 Prompt 质量（预生成评分）
  if (!input.videoUrl) {
    const score = evaluatePromptQuality(input);
    return {
      score,
      dimensions: { productCompleteness: 80, personConsistency: 80, shotFluency: 70, brandVisibility: 60, sellingPointCoverage: score, overallImpression: score },
      needsRetry: false,
      suggestions: [],
      rawJson: JSON.stringify({ mode: 'prompt-only', score }),
    };
  }

  try {
    const raw = await callLLM({
      systemPrompt: `你是AI视频质量评估专家。根据传递的上下文评估视频质量。JSON格式：
{"score":0-100,"productCompleteness":0-100,"personConsistency":0-100,"shotFluency":0-100,"brandVisibility":0-100,"sellingPointCoverage":0-100,"overallImpression":0-100,"needsRetry":false,"suggestions":[]}
评分 < 85 需要重试（仅限一次）。`,
      userPrompt: `商品：${input.vision.productType} | 卖点：${input.vision.sellingPoints.join('、')}
策略：${input.strategy.videoGoal} | ${input.strategy.platform}
分镜数：${input.director.totalShots}
Prompt样本：${input.promptEngine.finalPrompt.slice(0, 200)}…`,
      temperature: 0.2,
      responseFormat: 'json_object',
    });

    const j = parseJSON(raw) || {};
    const score = j.score || 80;
    return {
      score,
      dimensions: {
        productCompleteness: j.productCompleteness || score,
        personConsistency: j.personConsistency || score,
        shotFluency: j.shotFluency || score,
        brandVisibility: j.brandVisibility || score,
        sellingPointCoverage: j.sellingPointCoverage || score,
        overallImpression: j.overallImpression || score,
      },
      needsRetry: j.needsRetry || (score < 85),
      suggestions: j.suggestions || [],
      rawJson: raw,
    };
  } catch {
    const score = evaluatePromptQuality(input);
    return {
      score,
      dimensions: { productCompleteness: score, personConsistency: score, shotFluency: score, brandVisibility: score, sellingPointCoverage: score, overallImpression: score },
      needsRetry: score < 85,
      suggestions: [],
      rawJson: JSON.stringify({ fallback: true, score }),
    };
  }
}

function evaluatePromptQuality(input: QualityInput): number {
  let s = 60;
  const fp = input.promptEngine.finalPrompt.length;
  s += Math.min(fp * 0.01, 10);
  if (input.vision.sellingPoints.length >= 3) s += 10;
  if (input.director.totalShots >= 4) s += 8;
  if (input.vision.visualTags.length >= 3) s += 7;
  return Math.min(s, 95);
}

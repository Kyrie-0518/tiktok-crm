import { Router, Request, Response, NextFunction } from 'express';

// ─── 确保表存在 ───
let _tableEnsured = false;
function ensureTables() {
  if (_tableEnsured) return; _tableEnsured = true;
  const db = getDb();
  db.exec(`
    CREATE TABLE IF NOT EXISTS skiis_daily_logs (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      user_id INTEGER NOT NULL REFERENCES users(id),
      log_date TEXT NOT NULL,
      content TEXT DEFAULT '',
      structured_content TEXT DEFAULT '',
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP
    );
    CREATE INDEX IF NOT EXISTS idx_skiis_daily_user_date ON skiis_daily_logs(user_id, log_date);

    CREATE TABLE IF NOT EXISTS skiis_weekly_reports (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      user_id INTEGER NOT NULL REFERENCES users(id),
      week_start TEXT NOT NULL,
      week_end TEXT NOT NULL,
      summary TEXT DEFAULT '',
      data_analysis TEXT DEFAULT '',
      action_items TEXT DEFAULT '',
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP
    );
    CREATE INDEX IF NOT EXISTS idx_skiis_weekly_user ON skiis_weekly_reports(user_id, week_start);

    CREATE TABLE IF NOT EXISTS skiis_data_files (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      user_id INTEGER NOT NULL REFERENCES users(id),
      file_name TEXT DEFAULT '',
      file_path TEXT DEFAULT '',
      dimension TEXT DEFAULT '',
      analysis_result TEXT DEFAULT '',
      uploaded_at DATETIME DEFAULT CURRENT_TIMESTAMP
    );
    CREATE INDEX IF NOT EXISTS idx_skiis_data_user ON skiis_data_files(user_id);

    CREATE TABLE IF NOT EXISTS skiis_preferences (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      user_id INTEGER UNIQUE NOT NULL REFERENCES users(id),
      common_dimensions TEXT DEFAULT '["GMV","素材","广告"]',
      common_kpis TEXT DEFAULT '["GMV","ROI","转化率"]',
      preferred_frameworks TEXT DEFAULT '["漏斗分析"]',
      attention_threshold INTEGER DEFAULT 10,
      updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
    );
  `);
}

// ─── 辅助函数：获取用户角色 ───
function getUserRoleKey(req: Request): string {
  const user = req.user as JwtPayload;
  if (user.roleKey) return user.roleKey;
  try {
    const db = getDb();
    const row = db.prepare('SELECT r.role_key FROM users u LEFT JOIN roles r ON u.role_id = r.id WHERE u.id = ?').get(user.userId) as any;
    return row?.role_key || 'staff';
  } catch { return 'staff'; }
}

// ─── 文件上传配置 ───
const uploadDir = path.join(__dirname, '..', '..', 'data', 'skiis-uploads');
if (!fs.existsSync(uploadDir)) fs.mkdirSync(uploadDir, { recursive: true });

const storage = multer.diskStorage({
  destination: uploadDir,
  filename: (_req, file, cb) => {
    const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1e9);
    const ext = path.extname(file.originalname);
    cb(null, `${uniqueSuffix}${ext}`);
  },
});
const upload = multer({
  storage,
  limits: { fileSize: 20 * 1024 * 1024 },
  fileFilter: (_req, file, cb) => {
    const allowed = ['.xlsx', '.xls', '.csv', '.json'];
    const ext = path.extname(file.originalname).toLowerCase();
    cb(null, allowed.includes(ext));
  },
});

// 获取每日记录列表（按日期范围）
router.get('/daily-logs', authMiddleware, (req: Request, res: Response) => {
  ensureTables();
  const user = req.user as JwtPayload;
  const db = getDb();
  const { date_from, date_to } = req.query;
  const isDeveloper = getUserRoleKey(req) === 'developer';

  let where = isDeveloper ? 'WHERE 1=1' : 'WHERE dl.user_id = ?';
  const params: any[] = isDeveloper ? [] : [user.userId];
  if (date_from) { where += ' AND dl.log_date >= ?'; params.push(date_from); }
  if (date_to) { where += ' AND dl.log_date <= ?'; params.push(date_to); }

  const list = db.prepare(
    `SELECT dl.*, u.display_name FROM skiis_daily_logs dl
     LEFT JOIN users u ON dl.user_id = u.id
     ${where} ORDER BY dl.log_date DESC, dl.id DESC`
  ).all(...params);
  res.json({ list });
});

// 保存今日工作记录（同一天覆盖更新）
router.post('/daily-logs', authMiddleware, (req: Request, res: Response) => {
  ensureTables();
  const user = req.user as JwtPayload;
  const db = getDb();
  const { log_date, content } = req.body;
  if (!log_date) return res.status(400).json({ error: '日期必填' });
  if (!content) return res.status(400).json({ error: '工作内容不能为空' });

  const existing = db.prepare('SELECT id FROM skiis_daily_logs WHERE user_id = ? AND log_date = ?')
    .get(user.userId, log_date) as any;

  if (existing) {
    db.prepare("UPDATE skiis_daily_logs SET content = ?, created_at = datetime('now','localtime') WHERE id = ?")
      .run(content, existing.id);
    res.json({ id: existing.id, updated: true });
  } else {
    const result = db.prepare(
      `INSERT INTO skiis_daily_logs (user_id, log_date, content, created_at)
       VALUES (?, ?, ?, datetime('now','localtime'))`
    ).run(user.userId, log_date, content);
    res.json({ id: result.lastInsertRowid, updated: false });
  }
});

// 删除某日记录

  ensureTables();
  const user = req.user as JwtPayload;
  const db = getDb();
  const isDeveloper = getUserRoleKey(req) === 'developer';
  const where = isDeveloper ? 'id = ?' : 'id = ? AND user_id = ?';
  const params: any[] = isDeveloper ? [req.params.id] : [req.params.id, user.userId];
  const record = db.prepare(`SELECT id FROM skiis_daily_logs WHERE ${where}`).get(...params) as any;
  if (!record) return res.status(404).json({ error: '记录不存在或无权限' });
  db.prepare('DELETE FROM skiis_daily_logs WHERE id = ?').run(req.params.id);
  res.json({ success: true });
});

    .get(req.params.id, user.userId) as any;
  if (!record) return res.status(404).json({ error: '文件不存在' });

  const filePath = path.join(uploadDir, record.file_path);
  if (!fs.existsSync(filePath)) return res.status(404).json({ error: '文件已被删除' });

  try {
    // 尝试读取为文本（支持 CSV/JSON，Excel 暂不支持直接解析）
    const content = fs.readFileSync(filePath, 'utf-8').slice(0, 50000);
    res.json({ file_name: record.file_name, content, dimension: record.dimension });
  } catch {
    res.status(500).json({ error: '文件读取失败' });
  }
});

router.put('/preferences', authMiddleware, (req: Request, res: Response) => {
  ensureTables();
  const user = req.user as JwtPayload;
  const db = getDb();
  const { common_dimensions, common_kpis, preferred_frameworks, attention_threshold } = req.body;
  db.prepare(
    `INSERT INTO skiis_preferences (user_id, common_dimensions, common_kpis, preferred_frameworks, attention_threshold)
     VALUES (?, ?, ?, ?, ?)
     ON CONFLICT(user_id) DO UPDATE SET
       common_dimensions=excluded.common_dimensions,
       common_kpis=excluded.common_kpis,
       preferred_frameworks=excluded.preferred_frameworks,
       attention_threshold=excluded.attention_threshold,
       updated_at=datetime('now','localtime')`
  ).run(
    user.userId,
    JSON.stringify(common_dimensions || ['GMV', '素材', '广告']),
    JSON.stringify(common_kpis || ['GMV', 'ROI', '转化率']),
    JSON.stringify(preferred_frameworks || ['漏斗分析']),
    attention_threshold || 10
  );
  res.json({ success: true });
});

router.get('/weekly-summary', authMiddleware, (req: Request, res: Response) => {
  ensureTables();
  const user = req.user as JwtPayload;
  const db = getDb();
  const { week_start, week_end } = req.query;
  if (!week_start || !week_end) return res.status(400).json({ error: '请指定周报起止日期' });

  // 每日记录
  const dailyLogs = db.prepare(
    `SELECT * FROM skiis_daily_logs WHERE user_id = ? AND log_date >= ? AND log_date <= ? ORDER BY log_date`
  ).all(user.userId, week_start, week_end) as any[];

  // 本周上传的数据文件
  const dataFiles = db.prepare(
    `SELECT * FROM skiis_data_files WHERE user_id = ? AND uploaded_at >= ?||' 00:00:00' AND uploaded_at <= ?||' 23:59:59' ORDER BY uploaded_at`
  ).all(user.userId, week_start, week_end) as any[];

  res.json({ daily_logs: dailyLogs, data_files: dataFiles });
});

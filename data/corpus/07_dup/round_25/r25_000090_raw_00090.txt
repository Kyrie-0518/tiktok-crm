import React from 'react';
import { Modal, Button, Space } from 'antd';
import { ExclamationCircleOutlined, WarningOutlined } from '@ant-design/icons';

interface ConfirmModalProps {
  open: boolean;
  title: string;
  content: React.ReactNode;
  onConfirm: () => void | Promise<void>;
  onCancel: () => void;
  confirmText?: string;
  cancelText?: string;
  confirmType?: 'danger' | 'primary';
  loading?: boolean;
  width?: number;
  icon?: React.ReactNode;
}

export default function ConfirmModal({
  open,
  title,
  content,
  onConfirm,
  onCancel,
  confirmText = '确认',
  cancelText = '取消',
  confirmType = 'danger',
  loading = false,
  width = 420,
}: ConfirmModalProps) {
  const handleOk = async () => {
    await onConfirm();
  };

  return (
    <Modal
      open={open}
      title={title}
      onCancel={onCancel}
      width={width}
      footer={
        <Space>
          <Button onClick={onCancel} disabled={loading}>
            {cancelText}
          </Button>
          <Button
            type="primary"
            danger={confirmType === 'danger'}
            loading={loading}
            onClick={handleOk}
          >
            {confirmText}
          </Button>
        </Space>
      }
    >
      <div style={{ padding: '12px 0', fontSize: 14, lineHeight: 1.8 }}>
        {content}
      </div>
    </Modal>
  );
}

import React from 'react';
import { Typography } from 'antd';

const { Text } = Typography;

interface DataCardProps {
  /** KPI 标签（如 "总销售额"） */
  title: string;
  /** 核心数值（如 "RM 10,978"） */
  value: string | number;
  /** 趋势/副标题（如 "今日 +RM 500"） */
  trend?: string;
  /** 图标 ReactNode */
  icon?: React.ReactNode;
  /** 图标背景色 */
  iconBg?: string;
  /** 图标文字色 */
  iconColor?: string;
  /** 点击回调 */
  onClick?: () => void;
}

/**
 * 统一指标卡片 (KPI)
 * 
 * 规范: 12px圆角 | 白色背景 | 28px/700 数值 | hover translateY(-2px)
 * 
 * 用法:
 * ```tsx
 * <DataCard title="总销售额" value="RM 10,978" trend="今日 +RM 500" icon={<DollarOutlined />} iconColor="#4568FF" iconBg="#EDF0FF" />
 * ```
 */
export default function DataCard({ title, value, trend, icon, iconBg = '#EDF0FF', iconColor = 'var(--bo-primary)', onClick }: DataCardProps) {
  return (
    <div
      onClick={onClick}
      style={{
        background: 'var(--bo-card-bg)',
        borderRadius: 'var(--bo-radius-lg)',
        border: '1px solid var(--bo-border)',
        boxShadow: 'var(--bo-shadow-sm)',
        padding: '16px 20px',
        height: 120,
        display: 'flex', flexDirection: 'column', justifyContent: 'center',
        transition: 'all 0.2s',
        cursor: onClick ? 'pointer' : 'default',
      }}
      onMouseEnter={e => { e.currentTarget.style.transform = 'translateY(-2px)'; e.currentTarget.style.boxShadow = 'var(--bo-shadow-md)'; }}
      onMouseLeave={e => { e.currentTarget.style.transform = 'none'; e.currentTarget.style.boxShadow = 'var(--bo-shadow-sm)'; }}
    >
      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'flex-start' }}>
        <div style={{ flex: 1, minWidth: 0 }}>
          <Text style={{ fontSize: 12, color: 'var(--bo-text-secondary)' }}>{title}</Text>
          <div style={{
            fontSize: 'var(--bo-datacard-value-size)',
            fontWeight: 'var(--bo-datacard-value-weight)',
            color: 'var(--bo-text-title)',
            fontFamily: 'var(--bo-font-mono)',
            marginTop: 2,
            lineHeight: 1.2,
          }}>
            {value}
          </div>
          {trend && (
            <Text style={{ fontSize: 11, color: 'var(--bo-text-secondary)', display: 'block', marginTop: 2 }}>{trend}</Text>
          )}
        </div>
        {icon && (
          <div style={{
            width: 40, height: 40, borderRadius: 10,
            background: iconBg, color: iconColor,
            display: 'flex', alignItems: 'center', justifyContent: 'center',
            fontSize: 20, flexShrink: 0,
          }}>
            {icon}
          </div>
        )}
      </div>
    </div>
  );
}

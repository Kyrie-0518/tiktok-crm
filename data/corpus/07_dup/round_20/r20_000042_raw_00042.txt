import { Router, Request, Response } from 'express';
import getDb from '../../db';
import authMiddleware from '../../middleware/auth';
import { ORDER_DATE } from './shared';

import { calculateProductFinance } from './calc';

const router = Router();

// Date threshold constants
const PLATFORM_FEE_START = new Date('2026-06-05');
const COMMISSION_START = new Date('2026-07-25');

// Fixed rate constants
const RATE_PLATFORM_SUPPORT = 0.45;
const RATE_COMMISSION = 0.1026;
const RATE_TRANSACTION = 0.0378;
const RATE_BXP = 0.0486;

function parseOrderTime(orderTime: string): Date {
  if (!orderTime) return new Date();
  if (orderTime.includes('/')) {
    const parts = orderTime.split(' ');
    const dateParts = parts[0].split('/');
    if (dateParts.length === 3) {
      const day = parseInt(dateParts[0], 10);
      const month = parseInt(dateParts[1], 10) - 1;
      const year = parseInt(dateParts[2], 10);
      const timePart = parts[1] || '00:00:00';
      return new Date(year, month, day, ...timePart.split(':').map(Number));
    }
  }
  return new Date(orderTime);
}

function getSkuCostPrice(db: any, sku: string, specName: string, productId: number): number {
  if (sku) {
    const row = db.prepare('SELECT cost_price FROM product_skus WHERE sku_code = ? LIMIT 1').get(sku.trim()) as any;
    if (row?.cost_price) return row.cost_price;
    const row2 = db.prepare('SELECT cost_price FROM product_skus WHERE LOWER(sku_code) = LOWER(?) LIMIT 1').get(sku.trim()) as any;
    if (row2?.cost_price) return row2.cost_price;
  }
  if (specName) {
    const row3 = db.prepare('SELECT cost_price FROM product_skus WHERE LOWER(spec_name) = LOWER(?) LIMIT 1').get(specName.trim()) as any;
    if (row3?.cost_price) return row3.cost_price;
  }
  if (productId) {
    const row4 = db.prepare('SELECT cost_price FROM product_skus WHERE product_id = ? ORDER BY id LIMIT 1').get(productId) as any;
    if (row4?.cost_price) return row4.cost_price;
  }
  return 0;
}

export function calculateOrderProfit(order: any, orderItems: any[], exchangeRate: number, db: any): any {
  const orderDate = parseOrderTime(order.order_time);

  const netRevenue = (order.actual_amount || 0) - (order.taxes || 0) - (order.shipping_fee || 0);
  const x = netRevenue * exchangeRate;

  let y = 0;
  const skuCostBreakdown: any[] = [];
  for (const item of orderItems) {
    const costPrice = getSkuCostPrice(db, item.sku, item.spec_name, item.product_id);
    const itemCost = (item.quantity || 1) * costPrice;
    y += itemCost;
    skuCostBreakdown.push({ sku: item.sku, spec_name: item.spec_name, quantity: item.quantity, cost_price: costPrice, item_cost: Math.round(itemCost * 100) / 100 });
  }

  const feeRate = RATE_TRANSACTION + RATE_BXP;
  let z = (order.actual_amount || 0) * feeRate * exchangeRate;
  if (orderDate >= COMMISSION_START) z += (order.actual_amount || 0) * RATE_COMMISSION * exchangeRate;
  const a = orderDate >= PLATFORM_FEE_START ? RATE_PLATFORM_SUPPORT * exchangeRate : 0;

  let c = 0;
  let productWeight = 0;
  let matchedProductId = orderItems.length > 0 ? orderItems[0].product_id : null;
  const firstItem = orderItems.length > 0 ? orderItems[0] : null;

  // 多级产品匹配：精确 → SKU模糊 → 名称模糊 → 规格名
  if (!matchedProductId && firstItem?.sku) {
    // 精确匹配 sku_code
    const skuExact = db.prepare('SELECT product_id FROM product_skus WHERE LOWER(sku_code) = LOWER(?) LIMIT 1').get(firstItem.sku) as any;
    if (skuExact?.product_id) matchedProductId = skuExact.product_id;
    // 模糊匹配 sku_code（处理 "Pemberish Bersaguna x1" 这类 Variation 名）
    if (!matchedProductId) {
      const skuFuzzy = db.prepare('SELECT product_id FROM product_skus WHERE LOWER(sku_code) LIKE ? LIMIT 1').get('%' + firstItem.sku.toLowerCase().replace(/\s+/g, '%') + '%') as any;
      if (skuFuzzy?.product_id) matchedProductId = skuFuzzy.product_id;
    }
    // 按 spec_name 匹配
    if (!matchedProductId) {
      const bySpec = db.prepare('SELECT product_id FROM product_skus WHERE LOWER(spec_name) = LOWER(?) OR LOWER(spec_name) LIKE ? LIMIT 1').get(firstItem.sku, '%' + firstItem.sku.toLowerCase() + '%') as any;
      if (bySpec?.product_id) matchedProductId = bySpec.product_id;
    }
  }
  // 按产品名称匹配
  if (!matchedProductId && firstItem?.product_name) {
    const nameExact = db.prepare('SELECT id FROM products WHERE LOWER(name) = LOWER(?) LIMIT 1').get(firstItem.product_name) as any;
    if (nameExact) matchedProductId = nameExact.id;
    // 名称包含关系（去掉数量后缀如 "x1" 再匹配）
    if (!matchedProductId) {
      const cleanName = firstItem.product_name.replace(/\s*x\d+\s*$/i, '').trim();
      const nameFuzzy = db.prepare('SELECT id FROM products WHERE LOWER(name) LIKE ? OR LOWER(name) LIKE ? LIMIT 1').get(
        '%' + cleanName.toLowerCase() + '%',
        '%' + firstItem.product_name.toLowerCase().replace(/\s+/g, '%') + '%'
      ) as any;
      if (nameFuzzy) matchedProductId = nameFuzzy.id;
    }
  }

  if (matchedProductId) {
    const product = db.prepare('SELECT weight FROM products WHERE id = ?').get(matchedProductId) as any;
    if (product?.weight) { productWeight = product.weight; c = (productWeight * 0.015) * exchangeRate; }
  }

  const netProfit = x - y - z - a - c;

  return {
    x: Math.round(x * 100) / 100,
    y: Math.round(y * 100) / 100,
    z: Math.round(z * 100) / 100,
    a: Math.round(a * 100) / 100,
    c: Math.round(c * 100) / 100,
    net_profit: Math.round(netProfit * 100) / 100,
    fee_breakdown: {
      commission_applied: orderDate >= COMMISSION_START,
      platform_support_applied: orderDate >= PLATFORM_FEE_START,
      commission_rate: orderDate >= COMMISSION_START ? RATE_COMMISSION : 0,
      transaction_rate: RATE_TRANSACTION,
      bxp_rate: RATE_BXP,
      platform_support_rate: RATE_PLATFORM_SUPPORT,
    },
    sku_costs: skuCostBreakdown,
    product_weight_g: productWeight,
    net_revenue_myr: Math.round(netRevenue * 100) / 100,
    taxes: order.taxes || 0,
    shipping_fee: order.shipping_fee || 0,
    actual_amount: order.actual_amount || 0,
  };
}

// GET /order-profit/summary
router.get('/summary', authMiddleware, (req: Request, res: Response) => {
  const db = getDb();
  const { shop_id, date_from, date_to } = req.query;

  let where = 'WHERE o.status NOT IN (\'cancelled\', \'auto_cancelled\', \'refunded\')';
  const params: any[] = [];
  if (shop_id) { where += ' AND o.shop_id = ?'; params.push(Number(shop_id)); }
  if (date_from) { where += ` AND ${ORDER_DATE} >= ?`; params.push(date_from); }
  if (date_to) { where += ` AND ${ORDER_DATE} <= ?`; params.push(date_to); }

  const revenueRow = db.prepare(`SELECT COALESCE(SUM(o.actual_amount), 0) as total_revenue, COALESCE(SUM(o.taxes), 0) as total_taxes, COALESCE(SUM(o.shipping_fee), 0) as total_shipping FROM orders o ${where}`).get(...params) as any;

  const orders = db.prepare(`SELECT o.* FROM orders o ${where}`).all(...params) as any[];
  const exchangeRate = getExchangeRate(db);

  let totalNetProfit = 0, totalX = 0, totalY = 0, totalZ = 0, totalA = 0, totalC = 0, ordersWithProfit = 0;
  const getOrderItems = db.prepare('SELECT * FROM order_items WHERE order_id = ?');

  for (const order of orders) {
    const items = getOrderItems.all(order.id) as any[];
    const calc = calculateOrderProfit(order, items, exchangeRate, db);
    totalNetProfit += calc.net_profit; totalX += calc.x; totalY += calc.y;
    totalZ += calc.z; totalA += calc.a; totalC += calc.c || 0; ordersWithProfit++;
  }

  res.json({
    exchange_rate: exchangeRate, date_from: date_from || null, date_to: date_to || null,
    module1: { label: '成交价总额', value_myr: Math.round((revenueRow.total_revenue || 0) * 100) / 100, value_rmb: Math.round((revenueRow.total_revenue || 0) * exchangeRate * 100) / 100, currency_display: 'MYR' },
    module2: { label: '总净利润', value_rmb: Math.round(totalNetProfit * 100) / 100, value_myr: Math.round((totalNetProfit / exchangeRate) * 100) / 100, currency_display: 'RMB', breakdown: { x: Math.round(totalX * 100) / 100, y: Math.round(totalY * 100) / 100, z: Math.round(totalZ * 100) / 100, a: Math.round(totalA * 100) / 100, c: Math.round(totalC * 100) / 100 } },
    stats: { order_count: ordersWithProfit, total_revenue_myr: Math.round((revenueRow.total_revenue || 0) * 100) / 100, total_taxes_myr: Math.round((revenueRow.total_taxes || 0) * 100) / 100, total_shipping_myr: Math.round((revenueRow.total_shipping || 0) * 100) / 100 },
  });
});

// GET /order-profit/records
router.get('/records', authMiddleware, (req: Request, res: Response) => {
  const db = getDb();
  const { shop_id, date_from, date_to, page = '1', page_size = '20' } = req.query;

  let where = 'WHERE o.status NOT IN (\'cancelled\', \'auto_cancelled\', \'refunded\')';
  const params: any[] = [];
  if (shop_id) { where += ' AND o.shop_id = ?'; params.push(Number(shop_id)); }
  if (date_from) { where += ` AND ${ORDER_DATE} >= ?`; params.push(date_from); }
  if (date_to) { where += ` AND ${ORDER_DATE} <= ?`; params.push(date_to); }

  const countRow = db.prepare(`SELECT COUNT(*) as total FROM orders o ${where}`).get(...params) as any;
  const pageNum = parseInt(page as string) || 1;
  const pageSize = parseInt(page_size as string) || 20;

  const orders = db.prepare(`
    SELECT o.*, ts.name as shop_name
    FROM orders o LEFT JOIN tiktok_shops ts ON o.shop_id = ts.id
    ${where} ORDER BY o.order_time DESC LIMIT ? OFFSET ?
  `).all(...params, pageSize, (pageNum - 1) * pageSize) as any[];

  const exchangeRate = getExchangeRate(db);
  const getOrderItems = db.prepare('SELECT * FROM order_items WHERE order_id = ?');

  const enriched = orders.map(order => {
    const items = getOrderItems.all(order.id) as any[];
    const calc = calculateOrderProfit(order, items, exchangeRate, db);
    return { ...order, profit_data: calc };
  });

  res.json({ records: enriched, total: countRow.total || 0, page: pageNum, page_size: pageSize, exchange_rate: exchangeRate });
});

export default router;

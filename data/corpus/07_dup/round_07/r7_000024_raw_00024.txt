/**
 * 移动端 Token 认证
 * - 生成扫码登录的一次性 token
 * - token 兑换 JWT
 */
import { Router, Request, Response } from 'express';
import crypto from 'crypto';
import jwt from 'jsonwebtoken';
import getDb from '../db';

const router = Router();

function getJwtSecret(): string {
  const db = getDb();
  const row = db.prepare("SELECT value FROM settings WHERE key = 'jwt_secret'").get() as any;
  return row?.value || process.env.JWT_SECRET || 'bozone_default_secret';
}

// POST /api/auth/generate-mobile-token
// 生成移动端扫码登录token（30天有效）
router.post('/generate-mobile-token', (req: Request, res: Response) => {
  const db = getDb();
  const user = (req as any).user;
  if (!user?.userId) return res.status(401).json({ error: '请先登录' });

  const token = crypto.randomBytes(32).toString('hex');
  const expires = new Date(Date.now() + 30 * 24 * 60 * 60 * 1000).toISOString();

  // 存储到数据库
  db.prepare('DELETE FROM mobile_tokens WHERE user_id = ?').run(user.userId);
  try {
    db.prepare('CREATE TABLE IF NOT EXISTS mobile_tokens (user_id INTEGER PRIMARY KEY, token TEXT, expires_at TEXT)').run();
  } catch {}
  db.prepare('INSERT INTO mobile_tokens (user_id, token, expires_at) VALUES (?, ?, ?)').run(user.userId, token, expires);

  const mobileUrl = `${process.env.FRONTEND_URL || 'https://bvefdvp.cn'}/m/kyrie?token=${token}`;
  res.json({ success: true, token, url: mobileUrl, expires_at: expires });
});

// POST /api/auth/revoke-mobile-token
// 撤销移动端token
router.post('/revoke-mobile-token', (req: Request, res: Response) => {
  const db = getDb();
  const user = (req as any).user;
  if (!user?.userId) return res.status(401).json({ error: '请先登录' });
  db.prepare('DELETE FROM mobile_tokens WHERE user_id = ?').run(user.userId);
  res.json({ success: true });
});

// POST /api/auth/mobile-login
// 移动端token换取JWT（无需登录状态）
router.post('/mobile-login', (req: Request, res: Response) => {
  const db = getDb();
  const { mobile_token } = req.body;
  if (!mobile_token) return res.status(400).json({ error: '缺少 mobile_token' });

  const row = db.prepare('SELECT user_id, expires_at FROM mobile_tokens WHERE token = ?').get(mobile_token) as any;
  if (!row) return res.status(401).json({ error: '无效的token，请重新扫码' });
  if (new Date(row.expires_at).getTime() < Date.now()) {
    db.prepare('DELETE FROM mobile_tokens WHERE token = ?').run(mobile_token);
    return res.status(401).json({ error: 'token已过期，请重新扫码' });
  }

  const user = db.prepare(`
    SELECT u.id, u.username, u.nickname, u.status, r.role_key as role_key
    FROM users u LEFT JOIN roles r ON u.role_id = r.id
    WHERE u.id = ? AND u.status = 'active'
  `).get(row.user_id) as any;
  if (!user) return res.status(401).json({ error: '用户不存在或已禁用' });

  const payload = { userId: user.id, username: user.username, roleKey: user.role_key, source: 'mobile' };
  const token = jwt.sign(payload, getJwtSecret(), { expiresIn: '7d' });

  res.json({ success: true, token, user: { id: user.id, username: user.username, nickname: user.nickname, roleKey: user.role_key } });
});

export default router;

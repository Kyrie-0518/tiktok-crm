router.get('/ids', authMiddleware, (req: Request, res: Response) => {
  const db = getDb();
  const { status, shop_id, keyword } = req.query;

  let sql = `SELECT id FROM orders WHERE 1=1`;
  const params: any[] = [];

  if (status && status !== 'all') {
    sql += ' AND status = ?'; params.push(status);
  }
  if (shop_id) {
    sql += ' AND shop_id = ?'; params.push(shop_id);
  }
  if (keyword) {
    sql += ' AND (order_no LIKE ? OR buyer_name LIKE ?)';
    params.push(`%${keyword}%`, `%${keyword}%`);
  }

  const rows = db.prepare(sql).all(...params) as any[];
  res.json(rows.map(r => r.id));
});

router.delete('/batch', authMiddleware, (req: Request, res: Response) => {
  const db = getDb();
  const { status, shop_id, keyword } = req.body;

  // Build same WHERE clause
  let whereSql = `WHERE 1=1`;
  const params: any[] = [];

  if (status && status !== 'all') {
    whereSql += ' AND status = ?'; params.push(status);
  }
  if (shop_id) {
    whereSql += ' AND shop_id = ?'; params.push(shop_id);
  }
  if (keyword) {
    whereSql += ' AND (order_no LIKE ? OR buyer_name LIKE ?)';
    params.push(`%${keyword}%`, `%${keyword}%`);
  }

  // Get IDs first
  const ids = db.prepare(`SELECT id FROM orders ${whereSql}`).all(...params) as any[];
  if (ids.length === 0) {
    return res.json({ deleted: 0 });
  }

  // Delete order_items then orders
  const idList = ids.map(r => r.id);
  const placeholders = idList.map(() => '?').join(',');
  db.prepare(`DELETE FROM order_items WHERE order_id IN (${placeholders})`).run(...idList);
  const result = db.prepare(`DELETE FROM orders WHERE id IN (${placeholders})`).run(...idList);

  res.json({ deleted: result.changes });
});

router.get('/', authMiddleware, (req: Request, res: Response) => {
  const db = getDb();
  const { status, shop_id, keyword, page = '1', page_size = '20' } = req.query;

  let sql = `
    SELECT o.*,
           ts.name as shop_name,
           i.name as influencer_name
    FROM orders o
    LEFT JOIN tiktok_shops ts ON o.shop_id = ts.id
    LEFT JOIN influencers i ON o.influencer_id = i.id
    WHERE 1=1
  `;
  const params: any[] = [];

  if (status && status !== 'all') {
    sql += ' AND o.status = ?'; params.push(status);
  }
  if (shop_id) {
    sql += ' AND o.shop_id = ?'; params.push(shop_id);
  }
  if (keyword) {
    sql += ' AND (o.order_no LIKE ? OR o.buyer_name LIKE ?)';
    params.push(`%${keyword}%`, `%${keyword}%`);
  }

  // Count total
  const countSql = sql.replace(
    /SELECT o\.\*[\s\S]+?FROM orders o/,
    'SELECT COUNT(*) as total FROM orders o'
  );
  const countRow = db.prepare(countSql).get(...params) as any;

  sql += ' ORDER BY o.order_time DESC LIMIT ? OFFSET ?';
  const pageNum = parseInt(page as string) || 1;
  const pageSize = parseInt(page_size as string) || 20;
  params.push(pageSize, (pageNum - 1) * pageSize);

  const orders = db.prepare(sql).all(...params) as any[];

  // Fetch items for each order
  for (const o of orders) {
    o.items = db.prepare(`
      SELECT oi.*,
             p.image as product_image,
             ps.image as sku_image
      FROM order_items oi
      LEFT JOIN products p ON oi.product_id = p.id
      LEFT JOIN product_skus ps ON oi.product_sku_id = ps.id
      WHERE oi.order_id = ?
    `).all(o.id);
    if (o.items.length > 0) {
      console.log(`[orders] 订单 ${o.id} / ${o.order_no} 返回 ${o.items.length} 条商品明细`);
    }
  }

  // Fallback: for items missing images, try matching by spec_name/sku_code (fuzzy)
  const fallbackBySkuCode = db.prepare(`SELECT ps.image, p.image as p_image FROM product_skus ps JOIN products p ON ps.product_id = p.id WHERE ps.sku_code = ? LIMIT 1`);
  const fallbackBySkuLike = db.prepare(`SELECT ps.image, p.image as p_image FROM product_skus ps JOIN products p ON ps.product_id = p.id WHERE ps.sku_code LIKE ? LIMIT 1`);
  const fallbackBySpec = db.prepare(`SELECT ps.image, p.image as p_image FROM product_skus ps JOIN products p ON ps.product_id = p.id WHERE ps.spec_name = ? LIMIT 1`);
  const fallbackByName = db.prepare(`SELECT image FROM products WHERE name = ? LIMIT 1`);
  for (const o of orders) {
    for (const item of (o.items as any[])) {
      if (!item.sku_image && !item.product_image) {
        // Try match by order_items.sku → product_skus.sku_code
        if (item.sku) {
          const skuRow = fallbackBySkuCode.get(item.sku.trim()) as any;
          if (skuRow?.image) { item.sku_image = skuRow.image; continue; }
          if (skuRow?.p_image) { item.product_image = skuRow.p_image; continue; }
        }
        // Try match by order_items.spec_name → product_skus.sku_code (TikTok Malay spec names stored as sku_code)
        if (item.spec_name) {
          const specRow = fallbackBySkuCode.get(item.spec_name.trim()) as any;
          if (specRow?.image) { item.sku_image = specRow.image; continue; }
          if (specRow?.p_image) { item.product_image = specRow.p_image; continue; }
          // Fuzzy match: spec_name LIKE '%keyword%'
          const fuzzyRow = fallbackBySkuLike.get('%' + item.spec_name.trim() + '%') as any;
          if (fuzzyRow?.image) { item.sku_image = fuzzyRow.image; continue; }
          if (fuzzyRow?.p_image) { item.product_image = fuzzyRow.p_image; continue; }
        }
        // Fallback: match by product_name
        if (item.product_name) {
          const pRow = fallbackByName.get(item.product_name) as any;
          if (pRow?.image) item.product_image = pRow.image;
        }
      }
    }
  }

  res.json({ orders, total: countRow?.total || 0, page: pageNum, page_size: pageSize });
});

router.post('/', authMiddleware, (req: Request, res: Response) => {
  const {
    order_no, shop_id, buyer_name, buyer_phone,
    status = 'pending', payment_status = 'unpaid',
    logistics_status = '', tracking_no = '', carrier = '',
    item_total = 0, shipping_fee = 0, discount = 0, taxes = 0, actual_amount = 0,
    influencer_id, remark = '', order_time, ship_deadline,
    items = [],
  } = req.body;

  if (!order_no) return res.status(400).json({ error: '订单号必填' });

  const db = getDb();

  // Auto-fetch commission_rate from influencer
  let commission_rate = 0;
  if (influencer_id) {
    const inf = db.prepare('SELECT commission_rate FROM influencers WHERE id = ?').get(influencer_id) as any;
    if (inf) commission_rate = inf.commission_rate || 0;
  }

  try {
    const result = db.prepare(`
      INSERT INTO orders (order_no, shop_id, buyer_name, buyer_phone, status, payment_status,
        logistics_status, tracking_no, carrier, item_total, shipping_fee, discount, taxes, actual_amount,
        influencer_id, commission_rate, remark, order_time, ship_deadline)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
    `).run(
      order_no, shop_id || null, buyer_name, buyer_phone, status, payment_status,
      logistics_status, tracking_no, carrier, item_total, shipping_fee, discount, taxes, actual_amount,
      influencer_id || null, commission_rate, remark,
      order_time || new Date().toISOString(), ship_deadline || null
    );

    const orderId = result.lastInsertRowid;

    // Insert items
    const insertItem = db.prepare(`
      INSERT INTO order_items (order_id, product_id, product_sku_id, sku, product_name, spec_name, quantity, unit_price, subtotal)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
    `);
    for (const item of items) {
      insertItem.run(orderId, item.product_id || null, item.product_sku_id || null, item.sku || '', item.product_name || '',
        item.spec_name || '', item.quantity || 1, item.unit_price || 0, item.subtotal || 0);
      // Deduct stock from product_skus if sku_id exists, otherwise from products
      if (item.product_sku_id && item.quantity) {
        db.prepare('UPDATE product_skus SET stock = MAX(0, stock - ?) WHERE id = ?').run(item.quantity, item.product_sku_id);
      } else if (item.product_id && item.quantity) {
        db.prepare('UPDATE products SET stock = MAX(0, stock - ?) WHERE id = ?').run(item.quantity, item.product_id);
      }
    }

    res.json({ id: orderId, order_no });
  } catch (e: any) {
    if (e.message?.includes('UNIQUE')) return res.status(400).json({ error: '订单号已存在' });
    throw e;
  }
});

router.put('/:id', authMiddleware, (req: Request, res: Response) => {
  const db = getDb();
  const order = db.prepare('SELECT * FROM orders WHERE id = ?').get(req.params.id) as any;
  if (!order) return res.status(404).json({ error: '订单不存在' });

  const {
    status, payment_status, logistics_status, tracking_no, carrier,
    remark, ship_deadline, influencer_id,
  } = req.body;

  const prevStatus = order.status;

  // Auto-sync commission_rate if influencer_id changed
  let commission_rate = order.commission_rate;
  if (influencer_id !== undefined && influencer_id !== order.influencer_id) {
    if (influencer_id) {
      const inf = db.prepare('SELECT commission_rate FROM influencers WHERE id = ?').get(influencer_id) as any;
      commission_rate = inf?.commission_rate || 0;
    } else {
      commission_rate = 0;
    }
  }

  db.prepare(`
    UPDATE orders SET
      status = ?, payment_status = ?, logistics_status = ?,
      tracking_no = ?, carrier = ?, remark = ?,
      ship_deadline = ?, influencer_id = ?, commission_rate = ?
    WHERE id = ?
  `).run(
    status ?? order.status,
    payment_status ?? order.payment_status,
    logistics_status ?? order.logistics_status,
    tracking_no ?? order.tracking_no,
    carrier ?? order.carrier,
    remark ?? order.remark,
    ship_deadline ?? order.ship_deadline,
    influencer_id ?? order.influencer_id,
    commission_rate,
    req.params.id
  );

  // If cancelled/refunded → restore stock
  const isCancelling = (status === 'cancelled' || status === 'refunded') &&
    prevStatus !== 'cancelled' && prevStatus !== 'refunded';
  if (isCancelling) {
    const items = db.prepare('SELECT * FROM order_items WHERE order_id = ?').all(order.id) as any[];
    for (const item of items) {
      if (item.product_sku_id) {
        db.prepare('UPDATE product_skus SET stock = stock + ? WHERE id = ?').run(item.quantity, item.product_sku_id);
      } else if (item.product_id) {
        db.prepare('UPDATE products SET stock = stock + ? WHERE id = ?').run(item.quantity, item.product_id);
      }
    }
  }

  res.json({ success: true });
});

router.get('/meta/counts', authMiddleware, (req: Request, res: Response) => {
  const db = getDb();
  const { shop_id } = req.query;
  const whereShop = shop_id ? 'WHERE shop_id = ?' : '';
  const params: any[] = shop_id ? [shop_id] : [];

  const rows = db.prepare(
    `SELECT status, COUNT(*) as c FROM orders ${whereShop} GROUP BY status`
  ).all(...params) as any[];

  const counts: Record<string, number> = { all: 0 };
  for (const r of rows) {
    counts[r.status] = r.c;
    counts.all = (counts.all || 0) + r.c;
  }
  res.json(counts);
});

// POST /api/orders/sync-from-platform — 电商平台订单同步到 ERP
router.post('/sync-from-platform', authMiddleware, (req: Request, res: Response) => {
  const {
    source_platform, source_order_id, order_no, buyer_name, buyer_id,
    order_amount, currency, status,
    tracking_number, logistics_provider, shipping_address,
    order_time, pay_time, ship_time, complete_time,
    buyer_note, seller_note, items = [],
  } = req.body;

  if (!order_no && !source_order_id) {
    return res.status(400).json({ error: '订单编号必填' });
  }

  const db = getDb();

  // 按 source_order_id 或 order_no 去重
  const existing = db.prepare(
    `SELECT * FROM orders WHERE source_order_id = ? OR order_no = ?`
  ).get(source_order_id || '', order_no || '') as any;

  // 状态映射（平台状态 → ERP 状态）
  const statusMap: Record<string, string> = {
    pending_payment: 'pending_payment',
    processing: 'pending_shipment',
    shipped: 'shipped',
    delivered: 'completed',
    cancelled: 'cancelled',
    returned: 'refund',
    refund: 'refund',
    unknown: 'pending_shipment',
  };
  const erpStatus = statusMap[status] || status || 'pending_shipment';

  if (existing) {
    // 更新现有订单
    db.prepare(`
      UPDATE orders SET
        status=?, actual_amount=?,
        buyer_name=?, buyer_note=?,
        tracking_number=?, logistics_provider=?,
        ship_time=CASE WHEN ? != '' THEN COALESCE(?, ship_time) ELSE ship_time END,
        complete_time=CASE WHEN ? != '' THEN COALESCE(?, complete_time) ELSE complete_time END,
        last_synced_at=datetime('now','localtime')
      WHERE id=?
    `).run(
      erpStatus, order_amount ?? existing.actual_amount,
      buyer_name || existing.buyer_name,
      buyer_note || '',
      tracking_number || existing.tracking_number,
      logistics_provider || '',
      ship_time || '', ship_time,
      complete_time || '', complete_time,
      existing.id
    );

    // 更新订单商品明细
    if (items && items.length > 0) {
      db.prepare('DELETE FROM order_items WHERE order_id = ?').run(existing.id);
      for (const item of items) {
        db.prepare(`
          INSERT INTO order_items (order_id, product_source_id, product_name,
            sku, quantity, unit_price, subtotal, image)
          VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        `).run(
          existing.id,
          item.product_source_id || '',
          item.product_name || '',
          item.sku || '',
          item.quantity || 1,
          item.unit_price || 0,
          item.subtotal || 0,
          item.image || ''
        );
      }
    }

    return res.json({ success: true, created: false, id: existing.id, message: '订单已更新' });
  } else {
    // 创建新订单
    const result = db.prepare(`
      INSERT INTO orders (
        shop_id, source_platform, source_order_id, order_no,
        buyer_name, buyer_id, actual_amount, currency,
        status, tracking_number, logistics_provider, shipping_address,
        order_time, pay_time, ship_time, complete_time,
        buyer_note, seller_note,
        created_at, updated_at, last_synced_at
      ) VALUES (
        1, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?,
        datetime('now','localtime'), datetime('now','localtime'), datetime('now','localtime')
      )
    `).run(
      source_platform || '',
      source_order_id || '',
      order_no || '',
      buyer_name || '',
      buyer_id || '',
      order_amount || 0,
      currency || 'MYR',
      erpStatus,
      tracking_number || '',
      logistics_provider || '',
      shipping_address || '',
      order_time || '',
      pay_time || '',
      ship_time || '',
      complete_time || '',
      buyer_note || '',
      seller_note || ''
    );

    const orderId = result.lastInsertRowid;

    // 插入订单商品明细
    if (items && items.length > 0) {
      for (const item of items) {
        db.prepare(`
          INSERT INTO order_items (order_id, product_source_id, product_name,
            sku, quantity, unit_price, subtotal, image)
          VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        `).run(
          orderId,
          item.product_source_id || '',
          item.product_name || '',
          item.sku || '',
          item.quantity || 1,
          item.unit_price || 0,
          item.subtotal || 0,
          item.image || ''
        );
      }
    }

    return res.json({ success: true, created: true, id: orderId, message: '订单已创建' });
  }
});

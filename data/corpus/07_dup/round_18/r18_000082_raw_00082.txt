import { create } from 'zustand';
import api from '../api';

export interface CostItem {
  id: number;
  name: string;
  currency: string;
  is_fixed: number;
  is_active: number;
  value_format: string; // 'number' | 'percentage'
  formula: string; // Global formula expression (e.g. "重量 * 8")
  created_at?: string;
}

export interface FinancialRecord {
  id?: number;
  product_id: number;
  cost_detail: Record<string, number>;
  custom_cost_formulas: Record<string, string>; // Per-product cost formula overrides
  net_profit: number;
  total_investment: number;
  roi: number;
  gross_margin: number;
  breakeven_roi: number;
  product_name?: string;
  sku?: string;
  sell_price?: number;
  cost_price?: number;
  weight?: number;
  image?: string;
  supplier_name?: string;
  created_at?: string;
  [key: string]: any; // Dynamic formula results
}

export interface FormulaItem {
  name: string;          // Display name = variable alias (merged)
  expression: string;    // Formula expression
  format: 'number' | 'percentage';
  currency: string;      // Currency for number format: RMB / MYR / USD / none (pure number)
  sort_order: number;    // 1-99
}

interface FinanceState {
  costItems: CostItem[];
  records: FinancialRecord[];
  summary: { total_revenue: number; total_profit: number; total_investment: number; overall_roi: number };
  formulas: FormulaItem[];
  exchangeRate: number;
  loading: boolean;
  fetchCostItems: () => Promise<void>;
  fetchRecords: (params?: any) => Promise<void>;
  fetchSummary: (params?: any) => Promise<void>;
  fetchFormulas: () => Promise<void>;
  updateFormulas: (formulas: FormulaItem[]) => Promise<void>;
  validateFormulas: (formulas: FormulaItem[], costItems: CostItem[]) => string | null;
  fetchExchangeRate: () => Promise<void>;
  updateExchangeRate: (rate: number) => Promise<void>;
  recalculateAll: () => Promise<void>;
  createRecord: (data: any) => Promise<void>;
  updateRecord: (id: number, data: any) => Promise<void>;
  deleteRecord: (id: number) => Promise<void>;
  createCostItem: (data: any) => Promise<void>;
  updateCostItem: (id: number, data: any) => Promise<void>;
  deleteCostItem: (id: number) => Promise<void>;
  syncProducts: () => Promise<{ total_products: number; added: number }>;
  batchUpdateCost: (cost_detail: Record<string, number>) => Promise<number>;
  batchApplyFormulas: (cost_formulas: Record<string, string>) => Promise<number>;
  /** Frontend-only: preview formula calculation for a product without server call */
  previewCalculation: (
    product: { sell_price: number; cost_price: number; weight: number; [key: string]: any },
    costItems: CostItem[],
    formulas: FormulaItem[],
    exchangeRate: number,
    costDetail?: Record<string, number>,
  ) => { formulaResults: Record<string, number>; costItemValues: Record<string, { raw: number; ctx: number; isPct: boolean; currency: string }>; myrTotalRmb: number };
}

export const useFinanceStore = create<FinanceState>((set, get) => ({
  costItems: [],
  records: [],
  summary: { total_revenue: 0, total_profit: 0, total_investment: 0, overall_roi: 0 },
  formulas: [
    { name: '毛利率', expression: '(售价 - 采购成本) / 售价', format: 'percentage', currency: 'RMB', sort_order: 1 },
    { name: '保本ROI', expression: '采购成本 / (售价 - 采购成本)', format: 'percentage', currency: 'RMB', sort_order: 2 },
    { name: '总投入', expression: '采购成本 + MYR成本合计RMB', format: 'number', currency: 'RMB', sort_order: 3 },
    { name: '净利润', expression: '售价 - 总投入', format: 'number', currency: 'MYR', sort_order: 4 },
    { name: '整体ROI', expression: '净利润 / 总投入', format: 'percentage', currency: 'RMB', sort_order: 5 },
  ],
  exchangeRate: 1.55,
  loading: false,

  fetchCostItems: async () => {
    const { data } = await api.get('/finance/cost-items');
    set({ costItems: data });
  },

  fetchRecords: async (params?: any) => {
    set({ loading: true });
    const { data } = await api.get('/finance/records', { params });
    set({ records: data, loading: false });
  },

  fetchSummary: async (params?: any) => {
    const { data } = await api.get('/finance/summary', { params });
    set({ summary: data });
  },

  fetchFormulas: async () => {
    const { data } = await api.get('/finance/formulas');
    set({ formulas: data });
  },

  updateFormulas: async (formulas: FormulaItem[]) => {
    await api.put('/finance/formulas', formulas);
    set({ formulas });
    await get().recalculateAll();
  },

  validateFormulas: (formulas: FormulaItem[], costItems: CostItem[]): string | null => {
    // Check duplicate names
    const names = formulas.map(f => f.name.trim());
    const dupes = names.filter((n, i) => names.indexOf(n) !== i);
    if (dupes.length > 0) return `公式名称重复：${dupes.join('、')}`;

    // Check empty names/expressions
    for (const f of formulas) {
      if (!f.name.trim()) return '公式名称不能为空';
      if (!f.expression.trim()) return `公式「${f.name}」的表达式不能为空`;
    }

    // Build known variables set
    const baseVars = new Set([
      '售价', '采购成本', '产品重量', 'MYR成本合计RMB', 'MYR兑RMB汇率',
      ...costItems.map(ci => ci.name),
    ]);
    // Formulas sorted by sort_order, each formula name becomes a variable
    const sorted = [...formulas].sort((a, b) => a.sort_order - b.sort_order);
    const formulaNames = new Map<string, number>(); // name -> sort_order

    for (const f of sorted) {
      // Check formula name doesn't clash with base variables
      if (baseVars.has(f.name)) {
        return `公式名称「${f.name}」与基础变量冲突，请更换名称`;
      }
      formulaNames.set(f.name, f.sort_order);

      // Extract variable references from expression
      const refs = extractRefs(f.expression);
      for (const ref of refs) {
        if (baseVars.has(ref)) continue;
        if (!formulaNames.has(ref)) {
          return `公式「${f.name}」引用了不存在的变量「${ref}」`;
        }
        if (formulaNames.get(ref)! >= f.sort_order) {
          return `公式「${f.name}」引用了尚未计算的公式「${ref}」（引用顺序错误）`;
        }
      }
    }
    return null;
  },

  fetchExchangeRate: async () => {
    try {
      const { data } = await api.get('/finance/exchange-rate');
      // 后端返回多货币格式 { "MYR": 1.74 }，兼容旧格式 { "rate": 1.74 }
      const rate = data?.MYR || data?.rate;
      if (typeof rate === 'number' && isFinite(rate) && rate > 0) {
        set({ exchangeRate: rate });
      }
    } catch {}
  },

  updateExchangeRate: async (rate: number) => {
    const { data } = await api.put('/finance/exchange-rate', { rate });
    // 后端返回 { success, rates: { "MYR": 1.74 } }，兼容旧格式
    const newRate = data?.rates?.MYR || data?.rate || rate;
    if (typeof newRate === 'number' && isFinite(newRate) && newRate > 0) {
      set({ exchangeRate: newRate });
    }
    // Backend already recalculates all records; just refresh frontend data
    await get().fetchRecords();
    await get().fetchSummary();
  },

  recalculateAll: async () => {
    await api.post('/finance/records/recalculate');
    get().fetchRecords();
    get().fetchSummary();
  },

  createRecord: async (formData: any) => {
    await api.post('/finance/records', formData);
    get().fetchRecords();
    get().fetchSummary();
  },

  updateRecord: async (id: number, formData: any) => {
    await api.put(`/finance/records/${id}`, formData);
    get().fetchRecords();
    get().fetchSummary();
  },

  deleteRecord: async (id: number) => {
    await api.delete(`/finance/records/${id}`);
    get().fetchRecords();
    get().fetchSummary();
  },

  createCostItem: async (formData: any) => {
    await api.post('/finance/cost-items', formData);
    get().fetchCostItems();
  },

  updateCostItem: async (id: number, formData: any) => {
    await api.put(`/finance/cost-items/${id}`, formData);
    get().fetchCostItems();
  },

  deleteCostItem: async (id: number) => {
    await api.delete(`/finance/cost-items/${id}`);
    get().fetchCostItems();
  },

  syncProducts: async () => {
    const { data } = await api.post('/finance/records/sync-products');
    get().fetchRecords();
    get().fetchSummary();
    return data;
  },

  batchUpdateCost: async (cost_detail: Record<string, number>) => {
    const { data } = await api.put('/finance/records/batch-cost', { cost_detail });
    get().fetchRecords();
    get().fetchSummary();
    return data.updated;
  },

  batchApplyFormulas: async (cost_formulas: Record<string, string>) => {
    const { data } = await api.put('/finance/records/batch-cost', { cost_formulas });
    get().fetchRecords();
    get().fetchSummary();
    return data.updated;
  },

  previewCalculation: (
    product, costItems, formulas, exchangeRate, costDetail = {},
  ) => {
    const safeRate = (typeof exchangeRate === 'number' && isFinite(exchangeRate) && exchangeRate > 0)
      ? exchangeRate : 1.55;
    // Build base context
    const ctx: Record<string, number> = {
      售价: product.sell_price || 0,
      采购成本: product.cost_price || 0,
      产品重量: product.weight || 0,
      MYR兑RMB汇率: safeRate,
    };

    // Add cost items as variables (mirror server-side calculateProductFinance)
    let myrTotalRmb = 0;
    const sellPrice = product.sell_price || 0;
    const costItemValues: Record<string, { raw: number; ctx: number; isPct: boolean; currency: string }> = {};

    for (const ci of costItems) {
      let rawVal = costDetail[ci.id] || 0;
      // If cost item has a global formula, evaluate it (priority: formula > fixed value)
      if (ci.formula && ci.formula.trim()) {
        rawVal = evaluateFormulaFrontend(ci.formula, ctx);
      }
      const isPct = ci.value_format === 'percentage';
      const val = isPct
        ? Math.round(sellPrice * rawVal / 100 * 100) / 100
        : rawVal;
      if (ci.currency === 'MYR') {
        const rmbVal = Math.round(val * safeRate * 100) / 100;
        ctx[ci.name] = rmbVal;
        myrTotalRmb += rmbVal;
        costItemValues[ci.name] = { raw: rawVal, ctx: rmbVal, isPct, currency: ci.currency };
      } else {
        ctx[ci.name] = val;
        costItemValues[ci.name] = { raw: rawVal, ctx: val, isPct, currency: ci.currency };
      }
    }
    ctx['MYR成本合计RMB'] = Math.round(myrTotalRmb * 100) / 100;

    // Evaluate formulas in sort_order
    const sorted = [...formulas].sort((a, b) => a.sort_order - b.sort_order);
    const formulaResults: Record<string, number> = {};

    for (const f of sorted) {
      if (!f.name.trim() || !f.expression.trim()) continue;
      const val = evaluateFormulaFrontend(f.expression, ctx);
      formulaResults[f.name] = val;
      ctx[f.name] = val;
    }

    return { formulaResults, costItemValues, myrTotalRmb };
  },
}));

// Extract variable-like references from a formula expression
function extractRefs(expr: string): string[] {
  const matches = expr.match(/[\u4e00-\u9fffA-Za-z0-9]+/g) || [];
  return [...new Set(matches.filter(m => !/^\d+(\.\d+)?$/.test(m)))];
}

// Frontend formula evaluator (mirrors server-side evaluateFormula)
export function evaluateFormulaFrontend(expression: string, ctx: Record<string, number>): number {
  let expr = expression;
  const keys = Object.keys(ctx).sort((a, b) => b.length - a.length);
  for (const key of keys) {
    const regex = new RegExp(key.replace(/[.*+?^${}()|[\]\\]/g, '\\$&'), 'g');
    expr = expr.replace(regex, String(ctx[key]));
  }
  if (!/^[\d\s+\-*/().%]+$/.test(expr)) return 0;
  try {
    const result = new Function(`"use strict"; return (${expr})`)();
    return typeof result === 'number' && isFinite(result) ? Math.round(result * 100) / 100 : 0;
  } catch {
    return 0;
  }
}

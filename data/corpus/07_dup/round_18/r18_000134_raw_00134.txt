import React from 'react';
import { Tabs } from 'antd';
import {
  SettingOutlined, KeyOutlined, AuditOutlined,
  SafetyOutlined,
} from '@ant-design/icons';
import { useNavigate, useSearchParams } from 'react-router-dom';
import Settings from './Settings';
import UserPermissions from './UserPermissions';
import AuditLogs from './AuditLogs';
import AdminRoles from './admin/Roles';
import AdminAuditLogs from './admin/AuditLogs';
import { useAuthStore } from '../stores/authStore';

const TAB_ITEMS = [
  { key: 'config', label: '系统配置', icon: <SettingOutlined />, component: Settings, permKey: 'settings-config' },
  { key: 'permissions', label: '用户与权限', icon: <KeyOutlined />, component: UserPermissions, permKey: 'settings-permissions' },
  { key: 'audit', label: '操作日志', icon: <AuditOutlined />, component: AuditLogs },
  { key: 'roles', label: '角色与账号', icon: <SafetyOutlined />, component: AdminRoles, adminOnly: true },
  { key: 'global-audit', label: '全局日志', icon: <AuditOutlined />, component: AdminAuditLogs, adminOnly: true },
];

export default function SystemSettings() {
  const [searchParams, setSearchParams] = useSearchParams();
  const roleKey = useAuthStore((s) => s.roleKey);
  const isDevOrAdmin = roleKey === 'developer' || roleKey === 'manager';

  // 根据 URL 参数或默认选中第一个可用 tab
  const activeTab = searchParams.get('tab') || 'config';

  const handleTabChange = (key: string) => {
    setSearchParams({ tab: key });
  };

  // 过滤可见 tabs：管理员可见全部，普通用户只看前三个
  const visibleTabs = TAB_ITEMS.filter(tab => !tab.adminOnly || isDevOrAdmin);

  // 渲染当前选中的组件
  const ActiveComponent = TAB_ITEMS.find(t => t.key === activeTab)?.component
    || TAB_ITEMS[0].component;

  return (
    <div>
      <div style={{ marginBottom: 16 }}>
        <Tabs
          activeKey={activeTab}
          onChange={handleTabChange}
          type="card"
          size="middle"
          items={visibleTabs.map(tab => ({
            key: tab.key,
            label: (
              <span style={{ display: 'flex', alignItems: 'center', gap: 6 }}>
                {tab.icon}
                {tab.label}
              </span>
            ),
          }))}
        />
      </div>

      <div style={{ flex: 1 }}>
        <ActiveComponent />
      </div>
    </div>
  );
}

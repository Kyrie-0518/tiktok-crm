import React from 'react';
import { Typography } from 'antd';

const { Text } = Typography;

interface PageHeaderProps {
  /** 页面标题 */
  title: string;
  /** 页面副标题/描述 */
  description?: string;
  /** 标题左侧图标 */
  icon?: React.ReactNode;
  /** 右侧扩展区域（如按钮） */
  extra?: React.ReactNode;
  /** 自定义样式 */
  style?: React.CSSProperties;
}

/**
 * 统一页面标题组件
 * 
 * 规范: 24px/700 标题 | 14px/94A3B8 描述 | 渐变图标 | 间距 24px
 * 
 * 用法:
 * ```tsx
 * <PageHeader title="经营概览" description="店铺经营数据驾驶舱" icon={<BarChartOutlined />} />
 * ```
 */
export default function PageHeader({ title, description, icon, extra, style }: PageHeaderProps) {
  return (
    <div style={{
      display: 'flex', alignItems: 'center', justifyContent: 'space-between',
      marginBottom: 'var(--bo-space-lg)', ...style,
    }}>
      <div style={{ display: 'flex', alignItems: 'center', gap: 'var(--bo-space-sm)' }}>
        {icon && (
          <div style={{
            width: 40, height: 40, borderRadius: 'var(--bo-radius-lg)',
            background: `linear-gradient(135deg, #6B8CFF, var(--bo-primary))`,
            display: 'flex', alignItems: 'center', justifyContent: 'center',
            boxShadow: '0 4px 12px rgba(69,104,255,0.25)',
            flexShrink: 0,
          }}>
            {React.isValidElement(icon)
              ? React.cloneElement(icon as React.ReactElement<{ style?: React.CSSProperties }>, { style: { fontSize: 20, color: '#fff' } })
              : icon}
          </div>
        )}
        <div>
          <Text strong style={{
            fontSize: 'var(--bo-page-title-size)',
            fontWeight: 'var(--bo-page-title-weight)',
            color: 'var(--bo-text-title)',
            display: 'block',
            lineHeight: 1.3,
          }}>{title}</Text>
          {description && (
            <Text style={{
              fontSize: 'var(--bo-page-desc-size)',
              color: 'var(--bo-text-weak)',
              lineHeight: 1.4,
            }}>{description}</Text>
          )}
        </div>
      </div>
      {extra && <div style={{ flexShrink: 0 }}>{extra}</div>}
    </div>
  );
}

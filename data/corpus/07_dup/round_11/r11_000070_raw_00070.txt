/**
 * Diagnosis Engine — 调用 AI 生成诊断报告
 * 输入：规则结果 + 证据链 + 数据摘要
 * 输出：结构化诊断报告（问题、原因、影响、建议、优先级、置信度）
 */
import getDb from '../../db';
import { RuleResult, Evidence, UnifiedData, DiagnosisProblem, Suggestion, ActionItem } from './types';

export async function runDiagnosis(
  shopName: string,
  data: UnifiedData,
  rules: RuleResult[],
  evidenceList: Evidence[]
): Promise<{
  health_score: number;
  health_grade: string;
  executive_summary: string;
  problems: DiagnosisProblem[];
  suggestions: Suggestion[];
  action_plan: ActionItem[];
}> {
  // 健康评分：100 - 每条异常扣分
  const deductions: Record<string, number> = { P0: 15, P1: 8, P2: 4 };
  let healthScore = 100;
  for (const r of rules) {
    healthScore = Math.max(0, healthScore - deductions[r.severity]);
  }
  const grade = healthScore >= 85 ? 'A' : healthScore >= 70 ? 'B' : healthScore >= 55 ? 'C' : 'D';

  // 构建 Prompt
  const ruleText = rules.map(r => `- [${r.severity}] ${r.category}: ${r.title}（${r.description}）`).join('\n');
  const evidenceText = evidenceList.map(e =>
    `【${e.category}】之前: ${e.before} → 现在: ${e.after} | 影响: ${e.impact}`
  ).join('\n');
  const dataText = generateDataSummary(data);

  const systemPrompt = `你是一个 TikTok Shop 跨境电商运营专家。根据提供的店铺数据、异常检测结果和证据链，生成诊断报告。

请严格按以下 JSON 格式输出（不要输出其他内容）：

{
  "executive_summary": "一句话总结（50字以内）",
  "problems": [
    {
      "category": "traffic|conversion|inventory|logistics|pricing|ads|video",
      "severity": "P0|P1|P2",
      "title": "问题标题",
      "cause": "原因分析",
      "evidence": "引用证据（一句话）",
      "impact": "量化影响",
      "confidence": 85
    }
  ],
  "suggestions": [
    {
      "priority": "P0|P1|P2",
      "title": "建议标题",
      "expected_improvement": "预计提升（如 +28% CVR）",
      "time_cost": "预计耗时（如 1 天）",
      "difficulty": "低|中|高",
      "confidence": 90
    }
  ],
  "action_plan": [
    { "id": "act-1", "title": "待办事项", "priority": "P0|P1|P2" }
  ]
}

要求：
- 每个问题必须引用具体的证据数据
- 建议必须可执行、可量化
- 置信度基于数据充分程度评估
- 不要虚构不存在的数据`;

  const userPrompt = `店铺: ${shopName}
分析范围: ${data.meta.time_range_days} 天
数据统计: ${data.meta.total_products} 件商品 | ${data.meta.total_orders} 笔订单 | ${data.meta.total_ads} 条广告 | ${data.meta.total_videos} 个视频

=== 数据摘要 ===
${dataText}

=== 异常检测 ===
${ruleText || '未检测到显著异常'}

=== 证据链 ===
${evidenceText || '无异常证据'}`;

  try {
    const db = getDb();
    const channel = db.prepare("SELECT * FROM ai_channels WHERE is_default=1 AND status='enabled' ORDER BY priority LIMIT 1").get() as any;
    if (!channel) throw new Error('未配置 AI 通道');

    const controller = new AbortController();
    const timer = setTimeout(() => controller.abort(), 90000);

    const response = await fetch(channel.api_base, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${channel.api_key}` },
      body: JSON.stringify({
        model: channel.model || 'deepseek-v4-pro',
        messages: [
          { role: 'system', content: systemPrompt },
          { role: 'user', content: userPrompt },
        ],
        temperature: 0.3, max_tokens: 3000,
      }),
      signal: controller.signal,
    });
    clearTimeout(timer);

    if (!response.ok) throw new Error(`AI API: ${response.status}`);

    const json: any = await response.json();
    const content = json.choices?.[0]?.message?.content || '';

    // 解析 JSON
    let parsed: any;
    try {
      const match = content.match(/\{[\s\S]*\}/);
      parsed = match ? JSON.parse(match[0]) : {};
    } catch { parsed = {}; }

    return {
      health_score: healthScore,
      health_grade: grade,
      executive_summary: parsed.executive_summary || '店铺诊断完成',
      problems: (parsed.problems || []).map((p: any, i: number) => ({
        category: p.category || 'unknown',
        severity: p.severity || 'P2',
        title: p.title || '',
        cause: p.cause || '',
        evidence: p.evidence || '',
        impact: p.impact || '',
        confidence: p.confidence || 70,
      })),
      suggestions: (parsed.suggestions || []).map((s: any) => ({
        priority: s.priority || 'P2',
        title: s.title || '',
        expected_improvement: s.expected_improvement || '',
        time_cost: s.time_cost || '',
        difficulty: s.difficulty || '中',
        confidence: s.confidence || 75,
      })),
      action_plan: (parsed.action_plan || []).map((a: any, i: number) => ({
        id: a.id || `act-${i + 1}`,
        title: a.title || '',
        priority: a.priority || 'P2',
        completed: false,
        expected_impact: '',
      })),
    };
  } catch (e: any) {
    // AI 不可用时 fallback 到规则引擎结果
    return {
      health_score: healthScore,
      health_grade: grade,
      executive_summary: `基于 ${data.meta.total_products} 件商品、${data.meta.total_orders} 笔订单的自动化分析`,
      problems: rules.slice(0, 5).map(r => ({
        category: r.category, severity: r.severity,
        title: r.title, cause: r.description, evidence: '',
        impact: r.severity === 'P0' ? '高' : r.severity === 'P1' ? '中' : '低',
        confidence: 60,
      })),
      suggestions: [{
        priority: 'P1', title: '配置 AI 通道以获得完整诊断报告',
        expected_improvement: '—', time_cost: '—', difficulty: '低', confidence: 100,
      }],
      action_plan: [],
    };
  }
}

function generateDataSummary(data: UnifiedData): string {
  const lines: string[] = [];

  if (data.products.length > 0) {
    const prices = data.products.map(p => p.price).filter(Boolean);
    if (prices.length > 0) {
      lines.push(`商品价格区间: $${Math.min(...prices).toFixed(2)} ~ $${Math.max(...prices).toFixed(2)}`);
    }
    const categories = [...new Set(data.products.map(p => p.category).filter(Boolean))];
    if (categories.length > 0) lines.push(`类目: ${categories.slice(0, 5).join(', ')}`);
  }

  if (data.orders.length > 0) {
    const totalGMV = data.orders.reduce((s, o) => s + (o.payment_amount || 0), 0);
    const totalDiscount = data.orders.reduce((s, o) => s + (o.discount_amount || 0), 0);
    lines.push(`订单 GMV: $${totalGMV.toFixed(2)}`);
    lines.push(`优惠总额: $${totalDiscount.toFixed(2)}`);
    const aov = data.orders.length > 0 ? totalGMV / data.orders.length : 0;
    lines.push(`客单价: $${aov.toFixed(2)}`);
  }

  if (data.ads.length > 0) {
    const totalSpend = data.ads.reduce((s, a) => s + (a.spend || 0), 0);
    const totalGmv = data.ads.reduce((s, a) => s + (a.gmv || 0), 0);
    lines.push(`广告花费: $${totalSpend.toFixed(2)}, 广告 GMV: $${totalGmv.toFixed(2)}`);
  }

  if (data.reviews.length > 0) {
    const avgRating = data.reviews.reduce((s, r) => s + r.rating, 0) / data.reviews.length;
    const negative = data.reviews.filter(r => r.rating <= 2).length;
    lines.push(`平均评分: ${avgRating.toFixed(1)} / 5, 差评率: ${(negative / data.reviews.length * 100).toFixed(1)}%`);
  }

  return lines.join('\n') || '暂无详细数据';
}

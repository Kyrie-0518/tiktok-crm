import { useEffect, useState } from 'react';
import { Card, Table, Tag, Input, Select, Space, DatePicker, Typography, Statistic, Row, Col, Button, message, Modal, Form, InputNumber, Tooltip } from 'antd';
import {
  PayCircleOutlined, SearchOutlined, ReloadOutlined,
  FilterOutlined, PlusOutlined, ExportOutlined,
  CheckCircleOutlined, ClockCircleOutlined, WarningOutlined,
  FileTextOutlined, QuestionCircleOutlined,
} from '@ant-design/icons';
import dayjs, { Dayjs } from 'dayjs';
import api from '../api';

const { Text } = Typography;
const { RangePicker } = DatePicker;

interface AdBill {
  id: number;
  shop_name: string;
  ad_account: string;
  bill_type: string;
  amount: number;
  currency: string;
  status: string;
  period_start: string;
  period_end: string;
  due_date: string | null;
  paid_at: string | null;
  remark: string;
  created_at: string;
}

const STATUS_MAP: Record<string, { color: string; label: string; icon: React.ReactNode }> = {
  pending:   { color: 'orange', label: '待支付', icon: <ClockCircleOutlined /> },
  paid:      { color: 'green',  label: '已支付', icon: <CheckCircleOutlined /> },
  overdue:   { color: 'red',    label: '已逾期', icon: <WarningOutlined /> },
  partial:   { color: 'blue',   label: '部分付', icon: <FileTextOutlined /> },
};

const BILL_TYPE_OPTIONS = [
  { value: 'campaign', label: '广告投放' },
  { value: 'boost', label: '加热推广' },
  { value: 'affiliate', label: '联盟佣金' },
  { value: 'other', label: '其他费用' },
];

export default function AdBills() {
  const [bills, setBills] = useState<AdBill[]>([]);
  const [loading, setLoading] = useState(true);
  const [total, setTotal] = useState(0);
  const [page, setPage] = useState(1);
  const [pageSize] = useState(15);
  const [statusFilter, setStatusFilter] = useState<string | undefined>();
  const [typeFilter, setTypeFilter] = useState<string | undefined>();
  const [searchText, setSearchText] = useState('');
  const [dateRange, setDateRange] = useState<[dayjs.Dayjs, dayjs.Dayjs] | null>(null);
  const [stats, setStats] = useState({ total_amount: 0, pending_amount: 0, paid_amount: 0, overdue_count: 0 });
  const [addModalOpen, setAddModalOpen] = useState(false);
  const [form] = Form.useForm();

  const fetchBills = async () => {
    setLoading(true);
    try {
      const params: any = { page, page_size: pageSize };
      if (statusFilter) params.status = statusFilter;
      if (typeFilter) params.bill_type = typeFilter;
      if (searchText) params.keyword = searchText;
      if (dateRange) {
        params.start_date = dateRange[0].format('YYYY-MM-DD');
        params.end_date = dateRange[1].format('YYYY-MM-DD');
      }
      const res = await api.get('/ad-bills', { params });
      setBills(res.data.list || []);
      setTotal(res.data.total || 0);
    } catch (e: any) {
      message.error('加载广告账款失败: ' + (e.response?.data?.error || e.message));
    } finally { setLoading(false); }
  };

  const fetchStats = async () => {
    try {
      const res = await api.get('/ad-bills/stats');
      setStats(res.data);
    } catch {}
  };

  useEffect(() => { fetchBills(); }, [page, pageSize]);
  useEffect(() => { fetchStats(); }, []);

  const handleAdd = async (values: any) => {
    try {
      await api.post('/ad-bills', values);
      message.success('添加成功');
      setAddModalOpen(false);
      form.resetFields();
      fetchBills();
      fetchStats();
    } catch (e: any) {
      message.error(e.response?.data?.error || '添加失败');
    }
  };

  const columns = [
    {
      title: '店铺/账号',
      dataIndex: 'shop_name',
      key: 'shop',
      width: 150,
      render: (_v: string, r: AdBill) => (
        <div>
          <div style={{ fontWeight: 500 }}>{r.shop_name}</div>
          <Text type="secondary" style={{ fontSize: 11.5 }}>{r.ad_account}</Text>
        </div>
      ),
    },
    {
      title: '类型',
      dataIndex: 'bill_type',
      width: 100,
      render: (t: string) => {
        const opt = BILL_TYPE_OPTIONS.find(o => o.value === t);
        return <Tag>{opt?.label || t}</Tag>;
      },
    },
    {
      title: '金额',
      dataIndex: 'amount',
      width: 120,
      sorter: (a: AdBill, b: AdBill) => a.amount - b.amount,
      render: (a: number, r: AdBill) => (
        <span style={{ fontWeight: 600, fontSize: 14 }}>
          {r.currency === 'MYR' ? 'RM' : '¥'}{a.toLocaleString('zh-CN', { minimumFractionDigits: 2 })}
        </span>
      ),
    },
    {
      title: '账期',
      width: 160,
      render: (_: any, r: AdBill) => (
        <div style={{ fontSize: 12.5 }}>
          <div>{r.period_start} ~ {r.period_end}</div>
          {r.due_date && (
            <Text type="secondary">
              截止: {dayjs(r.due_date).format('YYYY-MM-DD')}
              {dayjs(r.due_date).isBefore(dayjs()) && r.status !== 'paid' &&
                <Tag color="red" style={{ marginLeft: 6, fontSize: 11 }}>逾期</Tag>}
            </Text>
          )}
        </div>
      ),
    },
    {
      title: '状态',
      dataIndex: 'status',
      width: 90,
      filters: Object.entries(STATUS_MAP).map(([k, v]) => ({ text: v.label, value: k })),
      onFilter: (val: any, record: AdBill) => record.status === val,
      render: (s: string) => {
        const st = STATUS_MAP[s];
        return st ? <Tag color={st.color} icon={st.icon}>{st.label}</Tag> : s;
      },
    },
    {
      title: '备注',
      dataIndex: 'remark',
      ellipsis: true,
      width: 140,
      render: (r: string) => r || '-',
    },
    {
      title: '操作',
      width: 80,
      render: (_: any, r: AdBill) => (
        <Space>
          {r.status === 'pending' && (
            <Button type="link" size="small" onClick={() => message.info('标记为已支付功能开发中')}>标记支付</Button>
          )}
          <Button type="link" size="small" danger>删除</Button>
        </Space>
      ),
    },
  ];

  return (
    <div>
      {/* 标题 */}
      <div style={{ marginBottom: 20, display: 'flex', alignItems: 'center', gap: 10 }}>
        <div style={{
          width: 36, height: 36, borderRadius: 10,
          background: 'linear-gradient(135deg, #f59e0b, #d97706)',
          display: 'flex', alignItems: 'center', justifyContent: 'center',
          boxShadow: '0 2px 8px rgba(245,158,11,0.3)',
        }}>
          <PayCircleOutlined style={{ color: '#fff', fontSize: 18 }} />
        </div>
        <div>
          <h2 style={{ margin: 0, fontSize: 18, fontWeight: 700, color: '#172033' }}>广告账款</h2>
          <Text type="secondary" style={{ fontSize: 13 }}>TikTok 广告费用账单与付款追踪</Text>
        </div>
      </div>

      {/* 统计卡片 */}
      <Row gutter={[16, 16]} style={{ marginBottom: 18 }}>
        <Col xs={12} sm={6}>
          <Card size="small" className="stat-card-top-border" style={{ borderRadius: 12 }}>
            <Statistic
              title={<Text type="secondary">总金额</Text>}
              prefix="$"
              value={stats.total_amount}
              precision={2}
              valueStyle={{ fontSize: 22, fontWeight: 700 }}
            />
          </Card>
        </Col>
        <Col xs={12} sm={6}>
          <Card size="small" style={{ borderTop: '3px solid #f59e0b', borderRadius: 12 }}>
            <Statistic
              title={<Text type="secondary">待支付</Text>}
              prefix={<WarningOutlined />}
              value={stats.pending_amount}
              precision={2}
              valueStyle={{ fontSize: 22, fontWeight: 700, color: '#d97706' }}
            />
          </Card>
        </Col>
        <Col xs={12} sm={6}>
          <Card size="small" style={{ borderTop: '3px solid #059669', borderRadius: 12 }}>
            <Statistic
              title={<Text type="secondary">已支付</Text>}
              prefix={<CheckCircleOutlined />}
              value={stats.paid_amount}
              precision={2}
              valueStyle={{ fontSize: 22, fontWeight: 700, color: '#059669' }}
            />
          </Card>
        </Col>
        <Col xs={12} sm={6}>
          <Card size="small" style={{ borderTop: '3px solid #dc2626', borderRadius: 12 }}>
            <Statistic
              title={<Text type="secondary">逾期笔数</Text>}
              value={stats.overdue_count}
              valueStyle={{ fontSize: 22, fontWeight: 700, color: '#dc2626' }}
            />
          </Card>
        </Col>
      </Row>

      {/* 筛选栏 */}
      <Card size="small" style={{ borderRadius: 12, marginBottom: 16, padding: '4px 0' }}>
        <Space wrap size={12}>
          <Input
            placeholder="搜索店铺/账号..."
            prefix={<SearchOutlined />}
            allowClear
            value={searchText}
            onChange={e => setSearchText(e.target.value)}
            onPressEnter={fetchBills}
            style={{ width: 180 }}
          />
          <Select
            placeholder="状态筛选"
            allowClear
            value={statusFilter}
            onChange={(v) => { setStatusFilter(v); setPage(1); }}
            style={{ width: 120 }}
          >
            {Object.entries(STATUS_MAP).map(([k, v]) => (
              <Select.Option key={k} value={k}>{v.label}</Select.Option>
            ))}
          </Select>
          <Select placeholder="类型筛选" allowClear value={typeFilter}
            onChange={(v) => { setTypeFilter(v); setPage(1); }} style={{ width: 130 }}>
            {BILL_TYPE_OPTIONS.map(o => <Select.Option key={o.value} value={o.value}>{o.label}</Select.Option>)}
          </Select>
          <RangePicker
            value={dateRange}
            onChange={(dates) => setDateRange(dates as [Dayjs, Dayjs] | null)}
            format="YYYY-MM-DD"
            placeholder={['开始日期', '结束日期']}
            size="middle"
          />
          <Button type="primary" icon={<SearchOutlined />} onClick={() => setPage(1)}>查询</Button>
          <Button icon={<ReloadOutlined />} onClick={() => { setSearchText(''); setStatusFilter(undefined); setTypeFilter(undefined); setDateRange(null); setPage(1); }}>重置</Button>
          <Button type="primary" icon={<PlusOutlined />} onClick={() => setAddModalOpen(true)} ghost>新增</Button>
          <Tooltip title="导出 Excel"><Button icon={<ExportOutlined />} /></Tooltip>
        </Space>
      </Card>

      {/* 表格 */}
      <Card size="small" style={{ borderRadius: 12, overflow: 'hidden' }}>
        <Table
          rowKey="id"
          dataSource={bills}
          columns={columns}
          loading={loading}
          pagination={{
            current: page, pageSize, total,
            showSizeChanger: false, showTotal: t => `共 ${t} 条`,
            onChange: p => setPage(p),
          }}
          scroll={{ x: 900 }}
        />
      </Card>

      {/* 新增弹窗 */}
      <Modal title="新增广告账款" open={addModalOpen}
        onCancel={() => setAddModalOpen(false)}
        onOk={() => form.submit()} okText="保存" cancelText="取消">
        <Form form={form} layout="vertical" onFinish={handleAdd}>
          <Form.Item name="shop_name" label="店铺名称" rules={[{ required: true }]}>
            <Input placeholder="请输入店铺名称" />
          </Form.Item>
          <Form.Item name="ad_account" label="广告账号" rules={[{ required: true }]}>
            <Input placeholder="TikTok Ads 账号" />
          </Form.Item>
          <Form.Item name="bill_type" label="账单类型" rules={[{ required: true }]}>
            <Select options={BILL_TYPE_OPTIONS} placeholder="选择类型" />
          </Form.Item>
          <Form.Item name="amount" label="金额" rules={[{ required: true }]}>
            <InputNumber style={{ width: '100%' }} min={0} precision={2} placeholder="0.00" prefix="$" />
          </Form.Item>
          <Form.Item name="currency" label="货币" initialValue="MYR">
            <Select options={[{value:'MYR',label:'马来西亚令吉(MYR)'},{value:'RMB',label:'人民币(RMB)'},{value:'USD',label:'美元(USD)'}]} />
          </Form.Item>
          <Form.Item name="period_start" label="账期开始" rules={[{ required: true }]}>
            <DatePicker format="YYYY-MM-DD" style={{ width: '100%' }} />
          </Form.Item>
          <Form.Item name="period_end" label="账期结束" rules={[{ required: true }]}>
            <DatePicker format="YYYY-MM-DD" style={{ width: '100%' }} />
          </Form.Item>
          <Form.Item name="remark" label="备注">
            <Input.TextArea rows={2} placeholder="可选备注信息" />
          </Form.Item>
        </Form>
      </Modal>
    </div>
  );
}

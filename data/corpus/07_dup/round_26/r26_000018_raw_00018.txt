/**
 * 企业微信 Bot 回调接口
 * 文档: https://developer.work.weixin.qq.com/document/path/90238
 *
 * 配置:
 * - WECOM_CORP_ID
 * - WECOM_AGENT_ID
 * - WECOM_SECRET
 * - WECOM_TOKEN
 * - WECOM_ENCODING_AES_KEY
 */
import { Router, Request, Response } from 'express';
import express from 'express';
import crypto from 'crypto';
import getDb from '../db';
import { getAvailableChannels } from './ai';
import { agentLoop } from './agent-chat';

const router = Router();

function isConfigured(): boolean {
  return !!(process.env.WECOM_CORP_ID && process.env.WECOM_SECRET && process.env.WECOM_TOKEN && process.env.WECOM_ENCODING_AES_KEY);
}

function sha1(s: string): string {
  return crypto.createHash('sha1').update(s).digest('hex');
}

// 解密企业微信消息
function decryptMsg(encryptedBody: string): string {
  const aesKey = Buffer.from(process.env.WECOM_ENCODING_AES_KEY + '=', 'base64');
  const decipher = crypto.createDecipheriv('aes-256-cbc', aesKey, aesKey.slice(0, 16));
  decipher.setAutoPadding(false);
  let decrypted = Buffer.concat([decipher.update(encryptedBody, 'base64'), decipher.final()]);
  // 去掉末尾补位
  const pad = decrypted[decrypted.length - 1];
  decrypted = decrypted.slice(0, decrypted.length - pad);
  // 跳过前16字节随机字符串 + 4字节msg_len
  const content = decrypted.slice(20).toString('utf-8');
  return content.slice(0, content.lastIndexOf('<'));
}

// 从 XML 中提取字段
function extractXmlField(xml: string, field: string): string {
  const re = new RegExp(`<${field}><!\\[CDATA\\[([^\\]]+)\\]\\]></${field}>`);
  const m = xml.match(re);
  return m ? m[1] : '';
}

// 获取 access_token
let cachedAccessToken = '';
let tokenExpiry = 0;
async function getAccessToken(): Promise<string> {
  if (cachedAccessToken && Date.now() < tokenExpiry) return cachedAccessToken;
  const url = `https://qyapi.weixin.qq.com/cgi-bin/gettoken?corpid=${process.env.WECOM_CORP_ID}&corpsecret=${process.env.WECOM_SECRET}`;
  const res = await fetch(url);
  const data = await res.json() as any;
  if (data.errcode !== 0) throw new Error(`获取企业微信token失败: ${data.errmsg}`);
  cachedAccessToken = data.access_token;
  tokenExpiry = Date.now() + (data.expires_in - 300) * 1000;
  return cachedAccessToken;
}

// 主动发送消息
async function sendMessage(userId: string, content: string): Promise<void> {
  const token = await getAccessToken();
  const url = `https://qyapi.weixin.qq.com/cgi-bin/message/send?access_token=${token}`;
  await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      touser: userId,
      msgtype: 'text',
      agentid: parseInt(process.env.WECOM_AGENT_ID || '0'),
      text: { content },
    }),
  });
}

// GET — URL 验证（需要配置才能解密）
router.get('/callback', (req: Request, res: Response) => {
  if (!isConfigured()) return res.status(503).json({ error: '企业微信未配置' });
  const { msg_signature, timestamp, nonce, echostr } = req.query as any;
  if (!echostr) return res.status(400).send('missing echostr');

  const token = process.env.WECOM_TOKEN!;
  const sorted = [token, timestamp, nonce, echostr].sort().join('');
  const sig = sha1(sorted);
  if (sig !== msg_signature) return res.status(403).send('signature mismatch');

  try {
    const aesKey = Buffer.from(process.env.WECOM_ENCODING_AES_KEY + '=', 'base64');
    const decipher = crypto.createDecipheriv('aes-256-cbc', aesKey, aesKey.slice(0, 16));
    decipher.setAutoPadding(false);
    let decrypted = Buffer.concat([decipher.update(echostr, 'base64'), decipher.final()]);
    const pad = decrypted[decrypted.length - 1];
    decrypted = decrypted.slice(0, decrypted.length - pad);
    const result = decrypted.slice(16).toString('utf-8');
    const len = result.readUInt32BE(0);
    const reply = result.slice(4, 4 + len).toString('utf-8');
    res.send(reply);
  } catch (e: any) {
    console.error('[WeCom] 验证解密失败:', e.message);
    res.status(500).send('decrypt failed');
  }
});

// POST — 接收消息
router.post('/callback', express.text({ type: ['text/xml', 'application/xml'], limit: '5mb' }), async (req: Request, res: Response) => {
  if (!isConfigured()) return res.status(503).json({ error: '企业微信未配置' });

  try {
    // 解析加密XML
    let body = '';
    if (typeof req.body === 'string') body = req.body;
    else if (Buffer.isBuffer(req.body)) body = req.body.toString('utf-8');
    else body = JSON.stringify(req.body);

    const encrypt = extractXmlField(body, 'Encrypt');
    if (!encrypt) {
      // 可能未加密，尝试直接解析
      console.log('[WeCom] 收到未加密消息:', body.slice(0, 200));
      res.status(200).send('success');
      return;
    }

    const decrypted = decryptMsg(encrypt);
    const msgType = extractXmlField(decrypted, 'MsgType');
    const fromUser = extractXmlField(decrypted, 'FromUserName');
    let content = extractXmlField(decrypted, 'Content');
    const recognition = extractXmlField(decrypted, 'Recognition');

    // 语音消息：平台已转文字
    if (msgType === 'voice' && recognition) {
      content = recognition;
    }

    if (!content) {
      res.status(200).send('success');
      return;
    }

    console.log(`[WeCom] ${fromUser}: ${content}`);

    // 调用 Agent
    const db = getDb();
    const channels = getAvailableChannels(db);
    if (channels.length === 0) {
      await sendMessage(fromUser, 'AI 服务暂不可用，请先在系统设置中配置 AI 模型。');
      res.status(200).send('success');
      return;
    }

    try {
      const result = await agentLoop(channels, content);
      const reply = result.report.length > 2000
        ? result.report.slice(0, 1990) + '\n\n[内容过长已截断，请登录PC查看完整报告]'
        : result.report;
      await sendMessage(fromUser, reply);
    } catch (e: any) {
      await sendMessage(fromUser, `处理失败：${e.message}`);
    }

    res.status(200).send('success');
  } catch (e: any) {
    console.error('[WeCom] 处理消息失败:', e.message);
    res.status(200).send('success'); // 仍返回success避免重试
  }
});

export default router;

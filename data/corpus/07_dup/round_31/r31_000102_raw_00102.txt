import React from 'react';
import { Tag as AntTag } from 'antd';

type TagVariant = 'success' | 'warning' | 'danger' | 'info' | 'default';

interface TagProps {
  variant?: TagVariant;
  children?: React.ReactNode;
  style?: React.CSSProperties;
  className?: string;
  onClick?: (e: React.MouseEvent<HTMLElement>) => void;
}

const TAG_STYLE: Record<TagVariant, { bg: string; color: string }> = {
  success:  { bg: 'var(--bo-success-bg)',  color: 'var(--bo-success)' },
  warning:  { bg: 'var(--bo-warning-bg)',  color: 'var(--bo-warning)' },
  danger:   { bg: 'var(--bo-danger-bg)',   color: 'var(--bo-danger)' },
  info:     { bg: 'var(--bo-primary-light)', color: 'var(--bo-primary)' },
  default:  { bg: '#F1F5F9',               color: 'var(--bo-text-secondary)' },
};

export default function Tag({ variant = 'default', children, style, ...rest }: TagProps) {
  const s = TAG_STYLE[variant];
  return (
    <AntTag
      {...rest}
      style={{
        borderRadius: 'var(--bo-radius-sm)',
        border: 'none',
        background: s.bg,
        color: s.color,
        fontSize: 11,
        fontWeight: 600,
        padding: '2px 8px',
        margin: 0,
        lineHeight: '18px',
        ...style,
      }}
    >
      {children}
    </AntTag>
  );
}

import React, { useState, useEffect } from 'react';
import {
  Card, Input, Button, Tag, Space, Spin, Select, Upload, message,
  Typography, Empty, Alert,
} from 'antd';
import {
  SendOutlined, RobotOutlined, UploadOutlined,
  SyncOutlined, WarningOutlined, BulbOutlined,
} from '@ant-design/icons';
import { useAIStore } from '../stores/aiStore';
import api from '../api';

const { TextArea } = Input;
const { Paragraph, Text } = Typography;

export default function AIAnalysis() {
  // Load AI config from server on mount
  const [configReady, setConfigReady] = useState(false);
  useEffect(() => {
    useAIStore.getState().loadConfig().then(() => setConfigReady(true));
  }, []);

  // -------- Section 1: Q&A --------
  const [question, setQuestion] = useState('');
  const [answer, setAnswer] = useState('');
  const [qaLoading, setQaLoading] = useState(false);

  // -------- Section 2: Report Interpretation --------
  const [reportType, setReportType] = useState('profit');
  const [interpreting, setInterpreting] = useState(false);
  const [interpretResult, setInterpretResult] = useState<{
    highlights: string; warnings: string; ranking: string; suggestions: string;
  } | null>(null);

  // -------- Section 3: Auto Processing --------
  const [processing, setProcessing] = useState(false);
  const [processResult, setProcessResult] = useState('');

  // ======== AI Call Helper (config read from server DB) ========
  const callAI = async (systemPrompt: string, userMessage: string): Promise<string> => {
    const { apiKey } = useAIStore.getState();
    if (!apiKey) {
      message.error('请先在「系统设置」页面配置 AI API Key');
      return '';
    }
    try {
      const res = await api.post('/ai/chat', {
        messages: [
          { role: 'system', content: systemPrompt },
          { role: 'user', content: userMessage },
        ],
        max_tokens: 2000,
      });
      return res.data.content || '(无返回内容)';
    } catch (e: any) {
      throw new Error(e.response?.data?.error || e.message || 'AI 调用失败');
    }
  };

  // ======== Section 1: Handle Ask ========
  const presetQuestions = [
    '本月利润最高的3个产品是哪些？',
    '当前有哪些产品处于亏损状态？',
    '达人对接情况如何，有哪些高潜力达人？',
    '整体ROI表现如何，有哪些优化建议？',
    'TikTok马来西亚市场最近什么品类好卖？',
    '新手做TikTok东南亚，有什么避坑建议？',
  ];

  const handleAsk = async () => {
    if (!question.trim()) return;
    setQaLoading(true);
    setAnswer('');
    try {
      const [productsRes, recordsRes, influencersRes] = await Promise.all([
        api.get('/products').catch(() => ({ data: [] })),
        api.get('/finance/records').catch(() => ({ data: [] })),
        api.get('/influencers').catch(() => ({ data: [] })),
      ]);
      const context = JSON.stringify({
        products: productsRes.data,
        financeRecords: recordsRes.data,
        influencers: influencersRes.data,
      }, null, 2);

      // 搜索知识库，获取相关知识点
      let knowledgeContext = '';
      try {
        const kbRes = await api.post('/ai/knowledge-search', { query: question, maxResults: 3 });
        if (kbRes.data?.results?.length > 0) {
          knowledgeContext = kbRes.data.results
            .map((r: any) => `【${r.title}】\n${r.content}`)
            .join('\n\n---\n\n');
        }
      } catch { /* 知识库不可用不影响主流程 */ }

      const systemPrompt = `你是博众智汇全域跨境经营管理系统的AI智能分析助手，具备跨境电商领域的专业知识。
      
当用户咨询以下业务数据时，请参考数据进行回答：
${context}

${knowledgeContext ? `以下是从知识库中检索到的相关专业知识，请优先参考这些知识来回答用户问题：\n${knowledgeContext}\n` : ''}
行为规则：
- 如果用户的提问与上述业务数据相关，请基于数据精准分析，突出关键数字和结论
- 如果用户的提问是跨境电商/TikTok运营等通用领域问题（如选品趋势、平台政策、运营技巧等），请基于知识库内容和你的专业知识自由回答，无需引用系统数据
- 如果知识库有相关内容，优先引用知识库的知识点
- 如果用户提问完全与跨境电商无关，请礼貌提示用户咨询业务相关问题
- 直接用中文回答，重点突出
- 回答尽量控制在500字以内`;
      const result = await callAI(systemPrompt, question);
      setAnswer(result);
      useAIStore.getState().addHistory(question, result);
    } catch (e: any) {
      message.error(e.message || '提问失败');
    } finally {
      setQaLoading(false);
    }
  };

  // ======== Section 2: Handle Interpret ========
  const handleInterpret = async () => {
    setInterpreting(true);
    setInterpretResult(null);
    try {
      let data: any = {};
      if (reportType === 'profit') {
        const [recordsRes, summaryRes] = await Promise.all([
          api.get('/finance/records').catch(() => ({ data: [] })),
          api.get('/finance/summary').catch(() => ({ data: {} })),
        ]);
        data = { records: recordsRes.data, summary: summaryRes.data };
      } else if (reportType === 'cost') {
        const [costItemsRes, recordsRes] = await Promise.all([
          api.get('/finance/cost-items').catch(() => ({ data: [] })),
          api.get('/finance/records').catch(() => ({ data: [] })),
        ]);
        data = { costItems: costItemsRes.data, records: recordsRes.data };
      } else if (reportType === 'influencer') {
        const [influencersRes] = await Promise.all([
          api.get('/influencers').catch(() => ({ data: [] })),
        ]);
        data = { influencers: influencersRes.data };
      }
      const reportName = reportType === 'profit' ? '利润' : reportType === 'cost' ? '成本' : '达人';
      const systemPrompt = `你是专业财务/运营分析专家。请对以下「${reportName}数据」进行解读，并按4个维度输出。

数据：
${JSON.stringify(data, null, 2)}

请按以下4个部分输出，每部分用【】包裹标题：
【盈亏亮点】
（盈利情况、关键数字、趋势判断）

【成本异常预警】
（异常成本项、潜在风险、需要关注的数据）

【TOP5排行】
（表现最好和最差的各TOP5，用具体数字说明）

【优化建议】
（具体可执行的3-5条建议）

规则：用中文回答，数字要具体，建议要可执行。`;
      const result = await callAI(systemPrompt, `请解读${reportName}报表，输出4个维度的分析`);
      // Parse the result into 4 sections
      const sections = parseInterpretation(result);
      setInterpretResult(sections);
    } catch (e: any) {
      message.error('解读失败：' + e.message);
    } finally {
      setInterpreting(false);
    }
  };

  // Parse AI output into 4 sections
  const parseInterpretation = (text: string) => {
    const getSection = (title: string) => {
      const regex = new RegExp(`【${title}】([\\s\\S]*?)(?=【|$)`);
      const match = text.match(regex);
      return match ? match[1].trim() : '';
    };
    let highlights = getSection('盈亏亮点');
    let warnings = getSection('成本异常预警');
    let ranking = getSection('TOP5排行');
    let suggestions = getSection('优化建议');
    // Fallback: if parsing failed, put everything in highlights
    if (!highlights && !warnings && !ranking && !suggestions) {
      highlights = text;
    }
    return { highlights, warnings, ranking, suggestions };
  };

  // ======== Section 3: Auto Processing ========
  const handleSyncProducts = async () => {
    setProcessing(true);
    setProcessResult('');
    try {
      const res = await api.post('/finance/records/sync-products');
      message.success(`同步成功！新增 ${res.data.added} 条产品记录，共 ${res.data.total_products} 条`);
      setProcessResult(`同步完成！新增 ${res.data.added} 条记录。`);
    } catch (e: any) {
      message.error(e.response?.data?.error || '同步失败');
    } finally {
      setProcessing(false);
    }
  };

  const handleDetectAnomalies = async () => {
    setProcessing(true);
    setProcessResult('');
    try {
      const [productsRes, recordsRes] = await Promise.all([
        api.get('/products').catch(() => ({ data: [] })),
        api.get('/finance/records').catch(() => ({ data: [] })),
      ]);
      const systemPrompt = `你是数据分析专家。分析以下业务数据，检测异常数据。
检测维度：
1. 售价为0或异常低（<1元）的产品
2. 采购成本为0的产品
3. 利润为0或负利润的产品数量占比
4. ROI异常（>1000% 或 < -100%）的记录
5. 有财务记录但没有绑定产品的异常情况

数据：
产品：${JSON.stringify(productsRes.data)}
财务记录：${JSON.stringify(recordsRes.data)}

请用中文输出，格式：
【异常概览】共发现X类异常，涉及Y条记录
【异常明细】逐条列出异常项（产品名、异常类型、具体数值）
【处理建议】针对每个异常给出处理建议`;
      const result = await callAI(systemPrompt, '请检测并分析报告中的异常数据');
      setProcessResult(result);
    } catch (e: any) {
      message.error('检测失败：' + e.message);
    } finally {
      setProcessing(false);
    }
  };

  const handleRecommendFormulas = async () => {
    setProcessing(true);
    setProcessResult('');
    try {
      const [costItemsRes, formulasRes] = await Promise.all([
        api.get('/finance/cost-items').catch(() => ({ data: [] })),
        api.get('/finance/formulas').catch(() => ({ data: [] })),
      ]);
      const systemPrompt = `你是跨境电商财务配置专家。基于现有配置，分析是否有遗漏的重要成本项或利润计算公式。

现有成本项：${JSON.stringify(costItemsRes.data)}
现有计算公式：${JSON.stringify(formulasRes.data)}

跨境电商TikTok小店常见的成本项包括：
- 平台佣金（通常5-6%）
- 达人佣金（10-20%）
- 物流/运费
- 广告费（TikTok Ads）
- 包材费
- 仓储费

请给出：
【建议新增成本项】列出名称、币种（RMB/MYR）、格式（数字/百分比）
【建议新增公式】列出名称、表达式、格式
【配置优化建议】对现有配置的调整建议`;
      const result = await callAI(systemPrompt, '请推荐成本项和公式配置');
      setProcessResult(result);
    } catch (e: any) {
      message.error('推荐失败：' + e.message);
    } finally {
      setProcessing(false);
    }
  };

  const handleParseExcel = (_file: any) => {
    message.info('AI解析Excel功能：请直接将Excel数据粘贴到问答区，AI可帮助分析');
    return false; // prevent upload
  };

  // ======== Render ========
  const hasAIConfig = !!useAIStore.getState().apiKey;

  return (
    <div style={{ padding: '20px 24px', background: '#f5f3f0', minHeight: '100%' }}>
      {/* 页面标题 */}
      <div style={{ display: 'flex', alignItems: 'center', gap: 10, marginBottom: 20 }}>
        <div style={{
          width: 36, height: 36, borderRadius: 10,
          background: 'linear-gradient(135deg, #4568FF, #6B8CFF)',
          display: 'flex', alignItems: 'center', justifyContent: 'center',
          color: '#fff', fontSize: 18,
        }}>
          <RobotOutlined />
        </div>
        <div>
          <h2 style={{ margin: 0, fontSize: 18, fontWeight: 700, color: '#172033' }}>AI 智能分析</h2>
          <span style={{ fontSize: 12, color: '#999' }}>数据问答 · 报表解读 · 自动处理</span>
        </div>
        {!hasAIConfig && (
          <Alert message="请先在「系统设置」页面配置 AI API Key" type="warning" showIcon style={{ marginLeft: 16, flex: 1 }} />
        )}
      </div>

      <Spin spinning={qaLoading || interpreting || processing || !configReady}>
        {/* Section 1 & 2 */}
        <div style={{ display: 'flex', gap: 16, marginBottom: 16, flexWrap: 'wrap' }}>
          {/* Section 1: Q&A */}
          <Card title="智能数据问答" style={{ flex: '1 1 400px', minWidth: 400 }}>
            <Space direction="vertical" style={{ width: '100%' }} size="middle">
              <div>
                <Text type="secondary" style={{ fontSize: 12, display: 'block', marginBottom: 4 }}>预设问题（点击填入）</Text>
                <Space wrap>
                  {presetQuestions.map(q => (
                    <Tag key={q} style={{ cursor: 'pointer', marginBottom: 4 }} onClick={() => setQuestion(q)}>
                      {q}
                    </Tag>
                  ))}
                </Space>
              </div>
              <TextArea
                value={question} onChange={e => setQuestion(e.target.value)}
                placeholder="输入你的问题，例如：本月利润最高的产品是哪些？"
                autoSize={{ minRows: 3, maxRows: 6 }}
              />
              <Button
                type="primary" icon={<SendOutlined />}
                onClick={handleAsk} loading={qaLoading}
                disabled={!question.trim() || !hasAIConfig}
              >
                提问
              </Button>
              {answer && (
                <div style={{
                  background: '#fafafa', borderRadius: 8,
                  padding: 16, marginTop: 8,
                  border: '1px solid #e8e8e8',
                }}>
                  <Paragraph style={{ whiteSpace: 'pre-wrap', marginBottom: 0 }}>{answer}</Paragraph>
                </div>
              )}
              {!answer && !qaLoading && (
                <Empty description="输入问题或点击预设问题开始提问" image={Empty.PRESENTED_IMAGE_SIMPLE} />
              )}
            </Space>
          </Card>

          {/* Section 2: Report Interpretation */}
          <Card title="报表智能解读" style={{ flex: '1 1 400px', minWidth: 400 }}>
            <Space direction="vertical" style={{ width: '100%' }} size="middle">
              <Select value={reportType} onChange={setReportType} style={{ width: '100%' }}>
                <Select.Option value="profit">产品利润表</Select.Option>
                <Select.Option value="cost">成本明细表</Select.Option>
                <Select.Option value="influencer">达人数据表</Select.Option>
              </Select>
              <Button
                type="primary" onClick={handleInterpret} loading={interpreting}
                disabled={!hasAIConfig}
              >
                开始解读
              </Button>
              {interpretResult && (
                <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: 12, marginTop: 8 }}>
                  <Card size="small" title="盈亏亮点" style={{ borderLeft: '3px solid #52c41a' }}>
                    <Paragraph style={{ whiteSpace: 'pre-wrap', marginBottom: 0, fontSize: 13 }}>
                      {interpretResult.highlights || '暂无数据'}
                    </Paragraph>
                  </Card>
                  <Card size="small" title="成本异常预警" style={{ borderLeft: '3px solid #faad14' }}>
                    <Paragraph style={{ whiteSpace: 'pre-wrap', marginBottom: 0, fontSize: 13 }}>
                      {interpretResult.warnings || '暂无数据'}
                    </Paragraph>
                  </Card>
                  <Card size="small" title="TOP5排行" style={{ borderLeft: '3px solid #4568FF' }}>
                    <Paragraph style={{ whiteSpace: 'pre-wrap', marginBottom: 0, fontSize: 13 }}>
                      {interpretResult.ranking || '暂无数据'}
                    </Paragraph>
                  </Card>
                  <Card size="small" title="优化建议" style={{ borderLeft: '3px solid #722ed1' }}>
                    <Paragraph style={{ whiteSpace: 'pre-wrap', marginBottom: 0, fontSize: 13 }}>
                      {interpretResult.suggestions || '暂无数据'}
                    </Paragraph>
                  </Card>
                </div>
              )}
              {!interpretResult && !interpreting && (
                <Empty description="选择报表类型后点击「开始解读」" image={Empty.PRESENTED_IMAGE_SIMPLE} />
              )}
            </Space>
          </Card>
        </div>

        {/* Section 3: Auto Processing */}
        <Card title="智能自动处理" style={{ marginBottom: 16 }}>
          <Space wrap size="middle">
            <Upload accept=".xlsx,.xls,.csv" showUploadList={false} beforeUpload={handleParseExcel}>
              <Button icon={<UploadOutlined />}>AI解析导入表格</Button>
            </Upload>
            <Button type="primary" icon={<SyncOutlined />} onClick={handleSyncProducts} loading={processing}>
              自动同步产品信息
            </Button>
            <Button danger icon={<WarningOutlined />} onClick={handleDetectAnomalies} loading={processing}>
              异常数据检测
            </Button>
            <Button icon={<BulbOutlined />} onClick={handleRecommendFormulas} loading={processing}>
              公式/成本项配置推荐
            </Button>
          </Space>
          {processResult && (
            <div style={{
              background: '#fafafa', borderRadius: 8,
              padding: 16, marginTop: 16,
              border: '1px solid #e8e8e8',
              maxHeight: 400, overflow: 'auto',
            }}>
              <Paragraph style={{ whiteSpace: 'pre-wrap', marginBottom: 0 }}>{processResult}</Paragraph>
            </div>
          )}
        </Card>
      </Spin>
    </div>
  );
}

// Helper: get current user's role_key
function getUserRoleKey(req: Request, db: any): string {
  const user = req.user as JwtPayload;
  if (user.roleKey) return user.roleKey;
  const row = db.prepare(`
    SELECT r.role_key FROM users u LEFT JOIN roles r ON u.role_id = r.id WHERE u.id = ?
  `).get(user.userId) as any;
  return row?.role_key || 'staff';
}

// GET /api/ai/config — read AI config from database (兼容旧接口)
router.get('/config', authMiddleware, (_req: Request, res: Response) => {
  const db = getDb();
  // 优先从 ai_channels 获取默认渠道
  const channel = db.prepare(`
    SELECT api_base, api_key, model FROM ai_channels 
    WHERE is_default = 1 AND status = 'enabled' AND api_key != ''
    ORDER BY priority ASC LIMIT 1
  `).get() as any;

  if (channel) {
    res.json({
      api_key: channel.api_key || '',
      api_base: channel.api_base || 'https://api.deepseek.com/v1',
      model: channel.model || 'deepseek-chat',
      source: 'ai_channels'
    });
  } else {
    // 兼容旧配置
    const row = db.prepare("SELECT value FROM settings WHERE key = ?").get('ai_config') as any;
    if (row) {
      try {
        const config = JSON.parse(row.value);
        res.json({ 
          api_key: config.api_key || '',
          api_base: config.api_base || 'https://api.deepseek.com/v1',
          model: config.model || 'deepseek-chat',
          source: 'settings'
        });
      } catch {
        res.json({ api_key: '', api_base: 'https://api.deepseek.com/v1', model: 'deepseek-chat', source: 'default' });
      }
    } else {
      res.json({ api_key: '', api_base: 'https://api.deepseek.com/v1', model: 'deepseek-chat', source: 'default' });
    }
  }
});

// POST /api/ai/config — save AI config to database (developer/manager only)
router.put('/config', authMiddleware, (req: Request, res: Response) => {
  const db = getDb();
  const roleKey = getUserRoleKey(req, db);
  const roleOrder: Record<string, number> = { staff: 0, manager: 1, developer: 2 };
  if (roleOrder[roleKey] < 1) {
    return res.status(403).json({ error: '仅管理员及以上可修改 AI 配置' });
  }
  const { api_key, api_base, model } = req.body;
  const value = JSON.stringify({ api_key: api_key || '', api_base: api_base || 'https://api.deepseek.com/v1', model: model || 'deepseek-chat' });
  db.prepare("INSERT OR REPLACE INTO settings (key, value) VALUES (?, ?)").run('ai_config', value);
  res.json({ success: true, message: '配置已保存' });
});

// ========== AI Chat Proxy (支持负载均衡和失败重试) ==========

export interface Channel {
  id: number;
  name: string;
  api_base: string;
  api_key: string;
  model: string;
  priority: number;
}

// 获取可用渠道列表（按优先级排序）
// 如果 ai_channels 表为空，回退到 settings 表读取旧版 ai_config
export function getAvailableChannels(db: any): Channel[] {
  const channels = db.prepare(`
    SELECT id, name, api_base, api_key, model, priority
    FROM ai_channels
    WHERE status = 'enabled' AND api_key != '' AND api_base != ''
    ORDER BY priority ASC, id ASC
  `).all() as Channel[];

  if (channels.length > 0) return channels;

  // 回退：从 settings 表读取旧版 ai_config
  const row = db.prepare("SELECT value FROM settings WHERE key = ?").get('ai_config') as any;
  if (row) {
    try {
      const config = JSON.parse(row.value);
      if (config.api_key && config.api_base) {
        return [{
          id: 0,
          name: '默认渠道（来自设置）',
          api_base: config.api_base,
          api_key: config.api_key,
          model: config.model || 'deepseek-chat',
          priority: 100,
        }];
      }
    } catch { /* ignore parse error */ }
  }
  return [];
}

// 更新渠道统计
export function updateChannelStats(db: any, channelId: number, success: boolean, latency: number, errorMsg?: string) {
  if (success) {
    db.prepare(`
      UPDATE ai_channels SET
        last_success_at = datetime('now', 'localtime'),
        last_used_at = datetime('now', 'localtime'),
        avg_latency = CASE 
          WHEN avg_latency = 0 THEN ?
          ELSE (avg_latency * success_count + ?) / (success_count + 1)
        END,
        success_count = success_count + 1,
        last_error_at = NULL,
        last_error_message = ''
      WHERE id = ?
    `).run(latency, latency, channelId);
  } else {
    db.prepare(`
      UPDATE ai_channels SET
        last_error_at = datetime('now', 'localtime'),
        last_error_message = ?,
        error_count = error_count + 1
      WHERE id = ?
    `).run((errorMsg || 'Unknown error').slice(0, 500), channelId);
  }
}

// 调用单个渠道
export async function callChannel(
  channel: Channel,
  messages: any[],
  max_tokens: number,
  preferredModel?: string,
  tools?: any[],
  toolChoice?: string
): Promise<{ success: boolean; data?: any; error?: string; latency?: number }> {
  const startTime = Date.now();
  const base = channel.api_base.replace(/\/+$/, '');
  const model = preferredModel || channel.model;

  try {
    const controller = new AbortController();
    const timer = setTimeout(() => controller.abort(), 120000);

    const body: any = {
      model,
      messages,
      max_tokens: max_tokens || 2000,
    };
    if (tools && tools.length > 0) {
      body.tools = tools;
      body.tool_choice = toolChoice || 'auto';
    }

    const response = await fetch(base, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${channel.api_key}`,
      },
      body: JSON.stringify(body),
      signal: controller.signal,
    });

    clearTimeout(timer);
    const latency = Date.now() - startTime;

    if (!response.ok) {
      const errText = await response.text().catch(() => '');
      let errMsg = `HTTP ${response.status}`;
      try {
        const errData = JSON.parse(errText);
        errMsg = errData.error?.message || errData.message || errMsg;
      } catch {}
      return { success: false, error: errMsg, latency };
    }

    const data = await response.json();
    return { success: true, data, latency };
  } catch (e: any) {
    const latency = Date.now() - startTime;
    if (e.name === 'AbortError') {
      return { success: false, error: '请求超时（120秒）', latency };
    }
    return { success: false, error: e.message || '未知错误', latency };
  }
}

// POST /api/ai/chat — AI chat proxy with load balancing and failover
router.post('/chat', authMiddleware, async (req: Request, res: Response) => {
  const { messages, max_tokens, model: preferredModel, channel_id } = req.body;

  if (!messages || !Array.isArray(messages) || messages.length === 0) {
    res.status(400).json({ error: 'messages 不能为空' });
    return;
  }

  const db = getDb();

  // 获取可用渠道
  let channels = getAvailableChannels(db);

  if (channels.length === 0) {
    res.status(400).json({ error: '未配置可用的 AI 渠道，请先在 AI 渠道管理中添加并启用渠道' });
    return;
  }

  // 如果指定了渠道 ID，只使用该渠道
  if (channel_id) {
    channels = channels.filter(ch => ch.id === channel_id);
    if (channels.length === 0) {
      res.status(400).json({ error: '指定的渠道不存在或未启用' });
      return;
    }
  }

  const errors: string[] = [];

  // 尝试调用每个渠道（负载均衡 + 失败重试）
  for (const channel of channels) {
    console.log(`[AI] 尝试渠道: ${channel.name} (${channel.id}), 模型: ${preferredModel || channel.model}`);

    const result = await callChannel(channel, messages, max_tokens || 2000, preferredModel);

    if (result.success && result.data) {
      // 更新统计
      updateChannelStats(db, channel.id, true, result.latency || 0);

      console.log(`[AI] 成功: ${channel.name}, 延迟: ${result.latency}ms`);

      // 返回时附加渠道信息
      return res.json({
        content: result.data.choices?.[0]?.message?.content || '(无返回内容)',
        channel: {
          id: channel.id,
          name: channel.name,
          latency: result.latency,
        },
        raw: result.data,
      });
    } else {
      // 记录错误
      errors.push(`${channel.name}: ${result.error}`);
      updateChannelStats(db, channel.id, false, result.latency || 0, result.error);

      console.log(`[AI] 失败: ${channel.name}, 错误: ${result.error}`);
    }
  }

  // 所有渠道都失败
  res.status(502).json({
    error: '所有 AI 渠道均调用失败',
    details: errors,
  });
});

// POST /api/ai/chat/stream — 流式 AI 对话 (支持负载均衡)
router.post('/chat/stream', authMiddleware, async (req: Request, res: Response) => {
  const { messages, max_tokens, model: preferredModel } = req.body;

  if (!messages || !Array.isArray(messages) || messages.length === 0) {
    res.status(400).json({ error: 'messages 不能为空' });
    return;
  }

  const db = getDb();
  const channels = getAvailableChannels(db);

  if (channels.length === 0) {
    res.status(400).json({ error: '未配置可用的 AI 渠道' });
    return;
  }

  // 流式只使用第一个可用渠道（默认渠道）
  const channel = channels[0];
  const base = channel.api_base.replace(/\/+$/, '');
  const model = preferredModel || channel.model;

  // 设置 SSE headers
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');

  try {
    const controller = new AbortController();

    const response = await fetch(base, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${channel.api_key}`,
      },
      body: JSON.stringify({
        model,
        messages,
        max_tokens: max_tokens || 2000,
        stream: true,
      }),
      signal: controller.signal,
    });

    if (!response.ok) {
      const errText = await response.text();
      res.status(response.status).json({ error: `AI 调用失败: ${errText}` });
      return;
    }

    // 转发流式响应（兼容 Web ReadableStream → Node stream）
    const reader = response.body?.getReader();
    if (reader) {
      const pump = async () => {
        while (true) {
          const { done, value } = await reader.read();
          if (done) {
            updateChannelStats(db, channel.id, true, 0);
            res.end();
            break;
          }
          res.write(value);
        }
      };
      pump().catch(() => res.end());
    }

  } catch (e: any) {
    updateChannelStats(db, channel.id, false, 0, e.message);
    if (!res.headersSent) {
      res.status(500).json({ error: e.message });
    }
  }
});

// ========== 知识库搜索 ==========

export const KB_DIR = path.resolve(__dirname, '../../docs/knowledge-base');
let knowledgeCache: { file: string; title: string; content: string; keywords: string[] }[] | null = null;
let lastCacheTime = 0;
const CACHE_TTL = 60_000; // 60秒后自动刷新，新增/修改的md文件自动生效

export function loadKnowledgeBase(): typeof knowledgeCache {
  const now = Date.now();
  // 缓存过期或首次加载 → 重新扫描所有md文件
  if (knowledgeCache && (now - lastCacheTime) < CACHE_TTL) return knowledgeCache;
  knowledgeCache = [];
  lastCacheTime = now;
  try {
    function walkDir(dir: string) {
      const entries = fs.readdirSync(dir, { withFileTypes: true });
      for (const entry of entries) {
        const fullPath = path.join(dir, entry.name);
        if (entry.isDirectory() && !entry.name.startsWith('.') && entry.name !== '行业报告与资料') {
          walkDir(fullPath);
        } else if (entry.isFile() && entry.name.endsWith('.md')) {
          const raw = fs.readFileSync(fullPath, 'utf-8');
          // Extract title (first # heading)
          const titleMatch = raw.match(/^#\s+(.+)/m);
          const title = titleMatch ? titleMatch[1] : entry.name.replace('.md', '');
          // Extract keywords
          const kwMatch = raw.match(/##\s+关键词\s*\n([\s\S]*?)(?=\n##|\n---|$)/);
          const keywords = kwMatch ? kwMatch[1].split(/[,，\n]/).map(k => k.trim()).filter(Boolean) : [];
          // Clean content (remove excessive markdown symbols for matching)
          const cleanContent = raw
            .replace(/^#{1,4}\s+/gm, '')
            .replace(/\*\*(.+?)\*\*/g, '$1')
            .replace(/`(.+?)`/g, '$1')
            .replace(/\[(.+?)\]\(.+?\)/g, '$1');
          knowledgeCache!.push({
            file: path.relative(KB_DIR, fullPath).replace(/\\/g, '/'),
            title,
            content: cleanContent,
            keywords,
          });
        }
      }
    }
    walkDir(KB_DIR);
    console.log(`[知识库] 已加载 ${knowledgeCache.length} 篇文章`);
  } catch (e) {
    console.error('[知识库] 加载失败:', e);
  }
  return knowledgeCache;
}

// POST /api/ai/knowledge-search — 搜索知识库
router.post('/knowledge-search', authMiddleware, (req: Request, res: Response) => {
  const { query, maxResults = 3 } = req.body;
  if (!query || typeof query !== 'string') {
    return res.status(400).json({ error: 'query 不能为空' });
  }

  const kb = loadKnowledgeBase();
  if (!kb || kb.length === 0) {
    return res.json({ results: [] });
  }

  // 分词：按空格、中英文标点拆分
  const terms = query
    .replace(/[，,。\.！!？?\s]+/g, ' ')
    .split(' ')
    .filter(t => t.length >= 2);

  // 计算每篇文章的匹配分
  const scored = kb.map(article => {
    let score = 0;
    const titleLower = article.title.toLowerCase();
    const contentLower = article.content.toLowerCase();

    for (const term of terms) {
      const termLower = term.toLowerCase();
      // 标题匹配权重 ×5
      const titleCount = (titleLower.match(new RegExp(termLower.replace(/[.*+?^${}()|[\]\\]/g, '\\$&'), 'g')) || []).length;
      score += titleCount * 5;
      // 关键词匹配权重 ×3
      for (const kw of article.keywords) {
        if (kw.toLowerCase().includes(termLower) || termLower.includes(kw.toLowerCase())) {
          score += 3;
        }
      }
      // 正文匹配权重 ×1
      const bodyCount = (contentLower.match(new RegExp(termLower.replace(/[.*+?^${}()|[\]\\]/g, '\\$&'), 'g')) || []).length;
      score += bodyCount;
    }

    return { ...article, score };
  });

  // 取 top N
  const topResults = scored
    .filter(r => r.score > 0)
    .sort((a, b) => b.score - a.score)
    .slice(0, maxResults)
    .map(r => ({
      title: r.title,
      file: r.file,
      content: r.content.slice(0, 3000), // 每篇最多3000字
      score: r.score,
      keywords: r.keywords,
    }));

  res.json({ results: topResults, query, totalIndexed: kb.length });
});
